001package sortpom.logger;
002
003/**
004 * @author bjorn
005 * @since 2012-12-21
006 */
007public interface SortPomLogger {
008    /**
009     * Send a message to the log in the <b>warn</b> error level.
010     *
011     * @param content warning message
012     */
013    void warn(String content);
014
015    /**
016     * Send a message to the log in the <b>info</b> error level.
017     *
018     * @param content info message
019     */
020    void info(String content);
021
022    /**
023     * Send a message to the log in the <b>error</b> error level.
024     *
025     * @param content error message
026     */
027    void error(String content);
028}
