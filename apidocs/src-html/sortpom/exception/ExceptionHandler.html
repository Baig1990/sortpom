001package sortpom.exception;
002
003import org.apache.maven.plugin.MojoFailureException;
004
005public final class ExceptionHandler {
006    private final FailureException fex;
007
008    public ExceptionHandler(FailureException fex) {
009        this.fex = fex;
010    }
011
012    public void throwMojoFailureException() throws MojoFailureException {
013        if (fex.getCause() != null) {
014            throw new MojoFailureException(fex.getMessage(), fex.getCause());
015        } else {
016            throw new MojoFailureException(fex.getMessage());
017        }
018    }
019}
