001package sortpom;
002
003import org.apache.commons.io.IOUtils;
004import org.jdom.Comment;
005import org.jdom.Document;
006import org.jdom.Element;
007import org.jdom.JDOMException;
008import org.jdom.input.SAXBuilder;
009import org.jdom.output.Format;
010import org.jdom.output.XMLOutputter;
011import sortpom.jdomcontent.NewlineText;
012import sortpom.parameter.LineSeparatorUtil;
013import sortpom.parameter.PluginParameters;
014import sortpom.util.BufferedLineSeparatorOutputStream;
015import sortpom.util.XmlOrderedResult;
016import sortpom.verify.ElementComparator;
017import sortpom.wrapper.operation.HierarchyWrapper;
018import sortpom.wrapper.operation.WrapperFactory;
019
020import java.io.ByteArrayOutputStream;
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.Writer;
024
025/**
026 * Creates xml structure and sorts it.
027 *
028 * @author Bjorn Ekryd
029 */
030public class XmlProcessor {
031    private final WrapperFactory factory;
032
033    private Document originalDocument;
034    private Document newDocument;
035    private String encoding;
036    private LineSeparatorUtil lineSeparatorUtil;
037    private String indentCharacters;
038    private boolean expandEmptyElements;
039    private boolean indentBlankLines;
040
041    public XmlProcessor(WrapperFactory factory) {
042        this.factory = factory;
043    }
044
045    /** Setup default configuration */
046    public void setup(PluginParameters pluginParameters) {
047        this.indentCharacters = pluginParameters.indentCharacters;
048        this.lineSeparatorUtil = pluginParameters.lineSeparatorUtil;
049        this.encoding = pluginParameters.encoding;
050        this.expandEmptyElements = pluginParameters.expandEmptyElements;
051        this.indentBlankLines = pluginParameters.indentBlankLines;
052    }
053
054    /**
055     * Sets the original xml that should be sorted. Builds a dom document of the
056     * xml.
057     *
058     * @param originalXml the new original xml
059     * @throws org.jdom.JDOMException the jDOM exception
060     * @throws java.io.IOException    Signals that an I/O exception has occurred.
061     */
062    public void setOriginalXml(final InputStream originalXml) throws JDOMException, IOException {
063        SAXBuilder parser = new SAXBuilder();
064        originalDocument = parser.build(originalXml);
065    }
066
067    /** Creates a new dom document that contains the sorted xml. */
068    public void sortXml() {
069        newDocument = (Document) originalDocument.clone();
070        final Element rootElement = (Element) originalDocument.getRootElement().clone();
071
072        HierarchyWrapper rootWrapper = factory.createFromRootElement(rootElement);
073
074        rootWrapper.createWrappedStructure(factory);
075        rootWrapper.detachStructure();
076        rootWrapper.sortStructureAttributes();
077        rootWrapper.sortStructureElements();
078        rootWrapper.connectXmlStructure();
079
080        newDocument.setRootElement(rootWrapper.getElementContent().getContent());
081    }
082
083    public XmlOrderedResult isXmlOrdered() {
084        ElementComparator elementComparator = new ElementComparator(originalDocument.getRootElement(), newDocument.getRootElement());
085        return elementComparator.isElementOrdered();
086    }
087
088    /**
089     * Returns the sorted xml as an OutputStream.
090     *
091     * @return the sorted xml
092     * @throws java.io.IOException
093     */
094    public ByteArrayOutputStream getSortedXml() throws IOException {
095        ByteArrayOutputStream sortedXml = new ByteArrayOutputStream();
096        BufferedLineSeparatorOutputStream bufferedLineOutputStream =
097                new BufferedLineSeparatorOutputStream(lineSeparatorUtil.toString(), sortedXml);
098
099        XMLOutputter xmlOutputter = new PatchedXMLOutputter(bufferedLineOutputStream, indentBlankLines);
100        xmlOutputter.setFormat(createPrettyFormat());
101        xmlOutputter.output(newDocument, bufferedLineOutputStream);
102
103        IOUtils.closeQuietly(bufferedLineOutputStream);
104        return sortedXml;
105    }
106
107    private Format createPrettyFormat() {
108        final Format prettyFormat = Format.getPrettyFormat();
109        prettyFormat.setExpandEmptyElements(expandEmptyElements);
110        prettyFormat.setEncoding(encoding);
111        prettyFormat.setLineSeparator("\n");
112        prettyFormat.setIndent(indentCharacters);
113        return prettyFormat;
114    }
115
116    private static class PatchedXMLOutputter extends XMLOutputter {
117        private final BufferedLineSeparatorOutputStream bufferedLineOutputStream;
118        private final boolean indentBlankLines;
119
120        public PatchedXMLOutputter(BufferedLineSeparatorOutputStream bufferedLineOutputStream, boolean indentBlankLines) {
121            this.bufferedLineOutputStream = bufferedLineOutputStream;
122            this.indentBlankLines = indentBlankLines;
123            XMLOutputter.preserveFormat.setLineSeparator("\n");
124        }
125
126        /** Stop XMLOutputter from printing comment <!-- --> chars if it is just a newline */
127        @Override
128        protected void printComment(Writer stringWriter, Comment comment) throws IOException {
129            if (comment instanceof NewlineText) {
130                if (!indentBlankLines) {
131                    clearIndentationForCurrentLine(stringWriter);
132                }
133            } else {
134                super.printComment(stringWriter, comment);
135            }
136        }
137
138        private void clearIndentationForCurrentLine(Writer stringWriter) throws IOException {
139            // Force all xml lines to be written to stream (via the writer)
140            stringWriter.flush();
141
142            // Remove all inset that has just been written since last newline
143            bufferedLineOutputStream.clearLineBuffer();
144        }
145    }
146}
