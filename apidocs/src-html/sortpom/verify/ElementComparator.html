001package sortpom.verify;
002
003import org.jdom.Element;
004import sortpom.util.XmlOrderedResult;
005
006import java.util.List;
007
008/**
009 * @author bjorn
010 * @since 2012-07-01
011 */
012public class ElementComparator {
013    private final Element originalElement;
014    private final Element newElement;
015
016    public ElementComparator(Element originalElement, Element newElement) {
017        this.originalElement = originalElement;
018        this.newElement = newElement;
019    }
020
021    private ElementComparator(Object originalElement, Object newElement) {
022        this.originalElement = (Element) originalElement;
023        this.newElement = (Element) newElement;
024    }
025
026    public XmlOrderedResult isElementOrdered() {
027        if (!originalElement.getName().equals(newElement.getName())) {
028            return XmlOrderedResult.nameDiffers(originalElement.getName(), newElement.getName());
029        }
030        if (isEqualsIgnoringWhitespace()) {
031            return XmlOrderedResult.textContentDiffers(originalElement.getName(), originalElement.getText(), newElement.getText());
032        }
033        return isChildrenOrdered(originalElement.getName(), originalElement.getChildren(), newElement.getChildren());
034    }
035
036    private boolean isEqualsIgnoringWhitespace() {
037        return !originalElement.getText().replaceAll("\\s", "").equals(newElement.getText().replaceAll("\\s", ""));
038    }
039
040    private XmlOrderedResult isChildrenOrdered(String name, List originalElementChildren, List newElementChildren) {
041        int size = Math.min(originalElementChildren.size(), newElementChildren.size());
042        for (int i = 0; i < size; i++) {
043            ElementComparator elementComparator = new ElementComparator(originalElementChildren.get(i), newElementChildren.get(i));
044            XmlOrderedResult elementOrdered = elementComparator.isElementOrdered();
045            if (!elementOrdered.isOrdered()) {
046                return elementOrdered;
047            }
048        }
049        if (originalElementChildren.size() != newElementChildren.size()) {
050            return XmlOrderedResult.childElementDiffers(name, originalElementChildren.size(), newElementChildren.size());
051        }
052        return XmlOrderedResult.ordered();
053    }
054}
