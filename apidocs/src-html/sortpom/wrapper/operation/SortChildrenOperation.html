001package sortpom.wrapper.operation;
002
003import org.jdom.Element;
004import sortpom.wrapper.content.Wrapper;
005
006import java.util.List;
007
008/**
009 * Xml hierarchy operation that all xml elements. Used by 
010 * Used in HierarchyWrapper.processOperation(HierarchyWrapperOperation operation)
011 * @author bjorn
012 * @since 2013-11-01
013 */
014public class SortChildrenOperation extends HierarchyWrapperOperation {
015    
016    /** Sort all children of an element */
017    @Override
018    public void manipulateChildElements(List<HierarchyWrapper> children) {
019        for (int i = 0; i < children.size(); i++) {
020            HierarchyWrapper wrapperImpl = children.get(i);
021            final Wrapper<Element> wrapper = wrapperImpl.getElementContent();
022
023            if (wrapper != null && wrapper.isSortable()) {
024                insertChildInSortedOrder(children, i, wrapperImpl, wrapper);
025            }
026        }
027
028    }
029
030    private void insertChildInSortedOrder(List<HierarchyWrapper> children, int indexOfChild, HierarchyWrapper wrapperImpl, Wrapper<Element> wrapper) {
031        boolean done = false;
032        for (int j = 0; !done && j < indexOfChild; j++) {
033            if (wrapper.isBefore(children.get(j).getElementContent())) {
034                children.remove(indexOfChild);
035                children.add(j, wrapperImpl);
036                done = true;
037            }
038        }
039    }
040}
