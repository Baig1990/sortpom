001package sortpom.wrapper.operation;
002
003import org.jdom.Attribute;
004import org.jdom.Element;
005import sortpom.wrapper.content.Wrapper;
006
007import java.util.ArrayList;
008import java.util.Collections;
009import java.util.List;
010
011/**
012 * Xml hierarchy operation that sort all attributes of xml elements. Used by 
013 * Used in HierarchyWrapper.processOperation(HierarchyWrapperOperation operation)
014 * @author bjorn
015 * @since 2013-11-01
016 */
017public class SortAttributesOperation extends HierarchyWrapperOperation {
018    private static final AttributeComparator ATTRIBUTE_COMPARATOR = new AttributeComparator();
019
020    /** Sort attributes of each element */
021    @Override
022    public void processElement(Wrapper<Element> elementWrapper) {
023        Element element = elementWrapper.getContent();
024        element.setAttributes(getSortedAttributes(element));
025    }
026
027    private List<Attribute> getSortedAttributes(Element element) {
028        final List<Attribute> attributes = getAttributeList(element);
029        for (Attribute attribute : attributes) {
030            attribute.detach();
031        }
032        Collections.sort(attributes, ATTRIBUTE_COMPARATOR);
033        return attributes;
034    }
035
036    @SuppressWarnings("unchecked")
037    private List<Attribute> getAttributeList(final Element element) {
038        return new ArrayList<Attribute>(element.getAttributes());
039    }
040
041}
