001package sortpom.wrapper;
002
003import org.jdom.Element;
004import sortpom.parameter.DependencySortOrder;
005import sortpom.parameter.PluginParameters;
006import sortpom.wrapper.content.*;
007
008import static sortpom.wrapper.ElementUtil.*;
009
010/**
011 * @author bjorn
012 * @since 2012-05-19
013 */
014public class ElementWrapperCreator {
015    private DependencySortOrder sortDependencies;
016    private DependencySortOrder sortPlugins;
017    private boolean sortProperties;
018    private final ElementSortOrderMap elementNameSortOrderMap;
019
020
021    public ElementWrapperCreator(ElementSortOrderMap elementNameSortOrderMap) {
022        this.elementNameSortOrderMap = elementNameSortOrderMap;
023    }
024
025    public void setup(PluginParameters pluginParameters) {
026        this.sortDependencies = pluginParameters.sortDependencies;
027        this.sortPlugins = pluginParameters.sortPlugins;
028        this.sortProperties = pluginParameters.sortProperties;
029    }
030
031    public Wrapper<Element> createWrapper(Element element) {
032        boolean sortedBySortOrderFile = elementNameSortOrderMap.containsElement(element);
033        if (sortedBySortOrderFile) {
034            if (isDependencyElement(element)) {
035                DependencySortedWrapper dependencySortedWrapper = new DependencySortedWrapper(element, elementNameSortOrderMap.getSortOrder(element));
036                dependencySortedWrapper.setSortOrder(sortDependencies);
037                return dependencySortedWrapper;
038            }
039            if (isPluginElement(element)) {
040                PluginSortedWrapper pluginSortedWrapper = new PluginSortedWrapper(element, elementNameSortOrderMap.getSortOrder(element));
041                pluginSortedWrapper.setSortOrder(sortPlugins);
042                return pluginSortedWrapper;
043            }
044            return new SortedWrapper(element, elementNameSortOrderMap.getSortOrder(element));
045        }
046        if (isPropertyElement(element)) {
047            return new AlphabeticalSortedWrapper(element);
048        }
049        return new UnsortedWrapper<Element>(element);
050    }
051
052    private boolean isDependencyElement(final Element element) {
053        if (sortDependencies.isNoSorting()) {
054            return false;
055        }
056        return isElementName(element, "dependency") && isElementParentName(element, "dependencies");
057    }
058
059    private boolean isPluginElement(final Element element) {
060        if (sortPlugins.isNoSorting()) {
061            return false;
062        }
063        if (isElementName(element, "plugin")) {
064            return isElementParentName(element, "plugins") || isElementParentName(element, "reportPlugins");
065        }
066        return false;
067    }
068
069    private boolean isPropertyElement(final Element element) {
070        if (!sortProperties) {
071            return false;
072        }
073        String deepName = getDeepName(element);
074        boolean inTheRightPlace = deepName.startsWith("/project/properties/")
075                || deepName.startsWith("/project/profiles/profile/properties/");
076        return inTheRightPlace && isElementParentName(element, "properties");
077    }
078
079}
