001package sortpom.wrapper.content;
002
003import org.jdom.Element;
004import sortpom.parameter.DependencySortOrder;
005
006import java.util.*;
007
008/**
009 * @author bjorn
010 * @since 2012-09-20
011 */
012public class ChildElementSorter {
013    public static final ChildElementSorter EMPTY_SORTER = new ChildElementSorter();
014    private final String[] childElementNames;
015    private final String[] childElementTexts;
016
017    public ChildElementSorter(DependencySortOrder dependencySortOrder, List<Element> children) {
018        this.childElementNames = dependencySortOrder.getChildElementNames();
019        this.childElementTexts = findChildElementTexts(children);
020    }
021
022    private ChildElementSorter() {
023        this.childElementNames = new String[0];
024        this.childElementTexts = new String[0];
025    }
026
027    private String[] findChildElementTexts(List<Element> children) {
028        Map<String, Element> childrenMap = getChildrenMappedByUpperCaseName(children);
029        List<String> childElementNameList = new ArrayList<String>();
030
031        for (String childElementName : childElementNames) {
032            childElementNameList.add(getChildText(childrenMap, childElementName));
033        }
034
035        return childElementNameList.toArray(new String[childElementNameList.size()]);
036    }
037
038    @SuppressWarnings("unchecked")
039    private Map<String, Element> getChildrenMappedByUpperCaseName(List<Element> children) {
040        Map<String, Element> map = new HashMap<String, Element>();
041
042        for (Element child : children) {
043            map.put(child.getName().toUpperCase(), child);
044        }
045
046        return map;
047    }
048
049    private String getChildText(Map<String, Element> children, String childElementName) {
050        Element element = children.get(childElementName.toUpperCase());
051        if (element == null) {
052            return "";
053        }
054        String text = element.getText();
055
056        return text == null ? "" : text;
057    }
058
059    public boolean compareTo(ChildElementSorter childElementSorter) {
060        if (childElementTexts.length != childElementSorter.childElementTexts.length) {
061            throw new IllegalStateException(String.format("This should not happen this: %s that: %s", toString(), childElementSorter.toString()));
062        }
063
064        return compareTexts(childElementSorter.childElementTexts) < 0;
065    }
066
067
068    private int compareTexts(String[] otherChildElementTexts) {
069        int compare;
070
071        for (int i = 0, childElementNamesLength = childElementNames.length; i < childElementNamesLength; i++) {
072            String childElementName = childElementNames[i];
073            String childElementText = childElementTexts[i];
074            String otherChildElementText = otherChildElementTexts[i];
075            if ("scope".equalsIgnoreCase(childElementName)) {
076                compare = compareScope(childElementText, otherChildElementText);
077            } else {
078                compare = childElementText.compareToIgnoreCase(otherChildElementText);
079            }
080
081            if (compare != 0) {
082                return compare;
083            }
084        }
085
086        return 0;
087    }
088
089    private int compareScope(String childElementText, String otherChildElementText) {
090        return Scope.getScope(childElementText).compareTo(Scope.getScope(otherChildElementText));
091    }
092
093    @Override
094    public String toString() {
095        return "ChildElementSorter{" +
096                "childElementTexts=" + (childElementTexts == null ? null : Arrays.asList(childElementTexts)) +
097                '}';
098    }
099
100    public void setEmptyPluginGroupIdValue(String value) {
101        for (int i = 0, childElementNamesLength = childElementNames.length; i < childElementNamesLength; i++) {
102            if ("groupId".equalsIgnoreCase(childElementNames[i]) && childElementTexts[i].isEmpty()) {
103                childElementTexts[i] = value;
104            }
105
106        }
107    }
108
109    private enum Scope {
110        COMPILE, PROVIDED, SYSTEM, RUNTIME, IMPORT, TEST, OTHER;
111
112        public static Scope getScope(String scope) {
113            if (scope == null || scope.isEmpty()) {
114                return COMPILE;
115            }
116            Scope[] values = Scope.values();
117            for (Scope value : values) {
118                if (scope.equalsIgnoreCase(value.name())) {
119                    return value;
120                }
121            }
122            return OTHER;
123        }
124    }
125}
