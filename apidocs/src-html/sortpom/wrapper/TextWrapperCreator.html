001package sortpom.wrapper;
002
003import org.jdom.Content;
004import org.jdom.Text;
005import sortpom.jdomcontent.NewlineText;
006import sortpom.parameter.PluginParameters;
007import sortpom.wrapper.content.ThrowAwayContentWrapper;
008import sortpom.wrapper.content.UnsortedWrapper;
009import sortpom.wrapper.content.Wrapper;
010
011/**
012 * @author bjorn
013 * @since 2012-05-19
014 */
015public class TextWrapperCreator {
016    private boolean keepBlankLines;
017
018    public TextWrapperCreator() {
019    }
020
021    public void setup(PluginParameters pluginParameters) {
022        keepBlankLines = pluginParameters.keepBlankLines;
023    }
024
025    public Wrapper<? extends Content> createWrapper(Text text) {
026        if (isSingleNewLine(text)) {
027            return ThrowAwayContentWrapper.INSTANCE;
028        } else if (isBlankLineOrLines(text)) {
029            return new UnsortedWrapper<Content>(new NewlineText());
030        }
031        return new UnsortedWrapper<Text>(text);
032    }
033
034    private boolean isSingleNewLine(Text content) {
035        return content.getText().matches("[\\t ]*[\\r]?[\\n]?[\\t ]*");
036    }
037
038    boolean isBlankLineOrLines(Text content) {
039        if (!keepBlankLines) {
040            return false;
041        }
042        return content.getText().matches("^\\s*?(\\r|\\n)\\s*$");
043    }
044}
