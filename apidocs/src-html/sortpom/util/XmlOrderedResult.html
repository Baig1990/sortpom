001package sortpom.util;
002
003/**
004 * This is the result returned from the verify operation. It contains a status and a error message (if applicable)
005 *
006 * @author bjorn
007 * @since 2012-08-11
008 */
009public final class XmlOrderedResult {
010    private final boolean ordered;
011    private final String errorMessage;
012
013    private XmlOrderedResult(boolean ordered, String errorMessage) {
014        this.ordered = ordered;
015        this.errorMessage = errorMessage;
016    }
017
018    /** pom file was ordered */
019    public static XmlOrderedResult ordered() {
020        return new XmlOrderedResult(true, "");
021    }
022
023    /** The xml elements was not in the right order */
024    public static XmlOrderedResult nameDiffers(String originalElementName, String newElementName) {
025        return new XmlOrderedResult(false, String.format("The xml element <%s> should be placed before <%s>",
026                newElementName, originalElementName));
027    }
028
029    /** The child elements of two elements differ. Example: When dependencies should be sorted */
030    public static XmlOrderedResult childElementDiffers(String name, int originalSize, int newSize) {
031        return new XmlOrderedResult(false, String.format(
032                "The xml element <%s> with %s child elements should be placed before element <%s> with %s child elements",
033                name, newSize, name, originalSize));
034    }
035
036    /** The texts inside two elements differ. Example: when maven properties should be sorted */
037    public static XmlOrderedResult textContentDiffers(String name, String originalElementText, String newElementText) {
038        return new XmlOrderedResult(false, String.format("The xml element <%s>%s</%s> should be placed before <%s>%s</%s>",
039                name, newElementText, name, name, originalElementText, name));
040    }
041
042    /** Returns true when verification tells that the pom file was sorted */
043    public boolean isOrdered() {
044        return ordered;
045    }
046
047    /** An error message that describes what went wrong with the verification operation */
048    public String getErrorMessage() {
049        return errorMessage;
050    }
051
052}
