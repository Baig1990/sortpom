001package sortpom.parameter;
002
003import java.io.File;
004
005/** Builder for the PluginParameters class */
006public class PluginParametersBuilder {
007    private File pomFile;
008    private boolean createBackupFile;
009    private String backupFileExtension;
010    private String encoding;
011    private LineSeparatorUtil lineSeparatorUtil;
012    private String indentCharacters;
013    private boolean indentBlankLines;
014    private boolean expandEmptyElements;
015    private String predefinedSortOrder;
016    private String customSortOrderFile;
017    private DependencySortOrder sortDependencies;
018    private DependencySortOrder sortPlugins;
019    private boolean sortProperties;
020    private boolean keepBlankLines;
021    private VerifyFailType verifyFailType;
022
023    /** Sets pomFile location */
024    public PluginParametersBuilder setPomFile(final File pomFile) {
025        this.pomFile = pomFile;
026        return this;
027    }
028
029    /** Sets information regarding backup file */
030    public PluginParametersBuilder setBackupInfo(final boolean createBackupFile, final String backupFileExtension) {
031        this.createBackupFile = createBackupFile;
032        this.backupFileExtension = backupFileExtension;
033        return this;
034    }
035
036    /** Sets which encoding should be used throughout the plugin */
037    public PluginParametersBuilder setEncoding(final String encoding) {
038        this.encoding = encoding;
039        return this;
040    }
041
042    /** Sets formatting information that is used when the pom file is sorted */
043    public PluginParametersBuilder setFormatting(final String lineSeparator,
044                                                 final boolean expandEmptyElements,
045                                                 final boolean keepBlankLines) {
046        this.lineSeparatorUtil = new LineSeparatorUtil(lineSeparator);
047        this.expandEmptyElements = expandEmptyElements;
048        this.keepBlankLines = keepBlankLines;
049        return this;
050    }
051
052    /** Sets indent information that is used when the pom file is sorted */
053    public PluginParametersBuilder setIndent(final int nrOfIndentSpace, final boolean indentBlankLines) {
054        this.indentCharacters = new IndentCharacters(nrOfIndentSpace).getIndentCharacters();
055        this.indentBlankLines = indentBlankLines;
056        return this;
057    }
058
059    /** Sets which sort order that should be used when sorting */
060    public PluginParametersBuilder setSortOrder(final String customSortOrderFile, final String predefinedSortOrder) {
061        this.customSortOrderFile = customSortOrderFile;
062        this.predefinedSortOrder = predefinedSortOrder;
063        return this;
064    }
065
066    /** Sets if any additional pom file elements should be sorted */
067    public PluginParametersBuilder setSortEntities(final String sortDependencies,
068                                                   final String sortPlugins, final boolean sortProperties) {
069        this.sortDependencies = new DependencySortOrder(sortDependencies);
070        this.sortPlugins = new DependencySortOrder(sortPlugins);
071        this.sortProperties = sortProperties;
072        return this;
073    }
074
075    /** Sets the verify operation behaviour */
076    public PluginParametersBuilder setVerifyFail(String verifyFail) {
077        this.verifyFailType = VerifyFailType.fromString(verifyFail);
078        return this;
079    }
080
081    /** Build the PluginParameters instance */
082    public PluginParameters createPluginParameters() {
083        return new PluginParameters(pomFile, createBackupFile, backupFileExtension,
084                encoding, lineSeparatorUtil, expandEmptyElements, keepBlankLines, indentCharacters, indentBlankLines,
085                predefinedSortOrder, customSortOrderFile,
086                sortDependencies, sortPlugins, sortProperties,
087                verifyFailType);
088    }
089}
