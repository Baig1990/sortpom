001package sortpom.parameter;
002
003import sortpom.exception.FailureException;
004
005import java.nio.charset.Charset;
006import java.util.Arrays;
007
008/**
009 * Encapsulates LineSeparation logic.
010 *
011 * @author bjorn
012 */
013public class LineSeparatorUtil {
014    private final String string;
015
016    /**
017     * Creates a line separator and makes sure that it is either &#92;n, &#92;r or &#92;r&#92;n
018     *
019     * @param lineSeparatorString The line separator characters
020     */
021    LineSeparatorUtil(final String lineSeparatorString) {
022        string = lineSeparatorString.replaceAll("\\\\r", "\r").replaceAll("\\\\n", "\n");
023        if (isIllegalString()) {
024            throw new FailureException(
025                    "LineSeparator must be either \\n, \\r or \\r\\n, but separator characters were "
026                            + Arrays.toString(lineSeparatorString.getBytes(Charset.defaultCharset())));
027        }
028    }
029
030    private boolean isIllegalString() {
031        return !("\n".equalsIgnoreCase(string) ||
032                "\r".equalsIgnoreCase(string) ||
033                "\r\n".equalsIgnoreCase(string));
034    }
035
036    @Override
037    public String toString() {
038        return string;
039    }
040
041}
