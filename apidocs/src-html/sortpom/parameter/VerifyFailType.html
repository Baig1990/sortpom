001package sortpom.parameter;
002
003import sortpom.exception.FailureException;
004
005/**
006 * @author bjorn
007 * @since 2012-08-11
008 */
009public enum VerifyFailType {
010    SORT, WARN, STOP;
011
012    static VerifyFailType fromString(String verifyFail) {
013        if (verifyFail == null) {
014            throw new FailureException("verifyFail must be either SORT, WARN or STOP. Was: null");
015        }
016        if ("SORT".equalsIgnoreCase(verifyFail)) {
017            return SORT;
018        }
019        if ("WARN".equalsIgnoreCase(verifyFail)) {
020            return WARN;
021        }
022        if ("STOP".equalsIgnoreCase(verifyFail)) {
023            return STOP;
024        }
025        throw new FailureException("verifyFail must be either SORT, WARN or STOP. Was: " + verifyFail);
026    }
027}
