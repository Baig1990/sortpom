001package sortpom.parameter;
002
003import java.io.File;
004
005/** Contains all parameters that are sent to the plugin */
006public class PluginParameters {
007    public final File pomFile;
008    public final boolean createBackupFile;
009    public final String backupFileExtension;
010    public final String encoding;
011    public final LineSeparatorUtil lineSeparatorUtil;
012    public final String indentCharacters;
013    public final boolean expandEmptyElements;
014    public final String predefinedSortOrder;
015    public final String customSortOrderFile;
016    public final DependencySortOrder sortDependencies;
017    public final DependencySortOrder sortPlugins;
018    public final boolean sortProperties;
019    public final boolean keepBlankLines;
020    public final boolean indentBlankLines;
021    public final VerifyFailType verifyFailType;
022
023    PluginParameters(File pomFile, boolean createBackupFile, String backupFileExtension, String encoding,
024                     LineSeparatorUtil lineSeparatorUtil, boolean expandEmptyElements, boolean keepBlankLines, String indentCharacters,
025                     boolean indentBlankLines, String predefinedSortOrder, String customSortOrderFile,
026                     DependencySortOrder sortDependencies, DependencySortOrder sortPlugins, boolean sortProperties, VerifyFailType verifyFailType) {
027        this.pomFile = pomFile;
028        this.createBackupFile = createBackupFile;
029        this.backupFileExtension = backupFileExtension;
030        this.encoding = encoding;
031        this.lineSeparatorUtil = lineSeparatorUtil;
032        this.indentCharacters = indentCharacters;
033        this.expandEmptyElements = expandEmptyElements;
034        this.predefinedSortOrder = predefinedSortOrder;
035        this.customSortOrderFile = customSortOrderFile;
036        this.sortDependencies = sortDependencies;
037        this.sortPlugins = sortPlugins;
038        this.sortProperties = sortProperties;
039        this.keepBlankLines = keepBlankLines;
040        this.indentBlankLines = indentBlankLines;
041        this.verifyFailType = verifyFailType;
042    }
043
044}
