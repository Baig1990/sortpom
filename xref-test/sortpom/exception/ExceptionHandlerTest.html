14  public class ExceptionHandlerTest {
15      @Rule
16      public ExpectedException expectedException = ExpectedException.none();
17      
18      @Test
19      public void failureExceptionShouldThrowMojoFailureException() throws MojoFailureException {
20          FailureException failureException = new FailureException("Gurka");
21          
22          expectedException.expect(MojoFailureException.class);
23          expectedException.expectMessage("Gurka");
24          
25          new ExceptionHandler(failureException).throwMojoFailureException();
26      }
27  
28      @Test
29      public void failureExceptionShouldKeepCause() throws MojoFailureException {
30          IllegalArgumentException cause = new IllegalArgumentException("not valid");
31          FailureException failureException = new FailureException("Gurka", cause);
32          
33          expectedException.expect(MojoFailureException.class);
34          expectedException.expectMessage("Gurka");
35          expectedException.expectCause(is(cause));
36          
37          new ExceptionHandler(failureException).throwMojoFailureException();
38      }
