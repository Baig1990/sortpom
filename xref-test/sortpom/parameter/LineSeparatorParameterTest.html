10  public class LineSeparatorParameterTest {

12      public ExpectedException thrown = ExpectedException.none();

14      @Test
15      public void lineSeparatorWithSomehtingElseShouldThrowException() throws Exception {
16          thrown.expect(FailureException.class);
17          thrown.expectMessage("LineSeparator must be either \\n, \\r or \\r\\n, but separator characters were [42, 42, 42]");
18  
19          new PluginParametersBuilder()
20                  .setEncoding("UTF-8")
21                  .setFormatting("***", false, false)
22                  .setIndent(2, false);
23  
24      }
25  
26      @Test
27      public void testCharInput() {
28          PluginParametersBuilder builder = new PluginParametersBuilder();
29          assertEquals("\n", builder.setFormatting("\n", true, true).createPluginParameters().lineSeparatorUtil.toString());
30          assertEquals("\r", builder.setFormatting("\r", true, true).createPluginParameters().lineSeparatorUtil.toString());
31          assertEquals("\r\n", builder.setFormatting("\r\n", true, true).createPluginParameters().lineSeparatorUtil.toString());
32      }
33  
34      @Test
35      public void testFailedInput1() {

37  
38          PluginParametersBuilder builder = new PluginParametersBuilder();
39          builder.setFormatting("\nn", true, true);
40      }

42      @Test
43      public void testFailedInput2() {
