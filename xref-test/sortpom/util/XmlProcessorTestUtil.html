27  public class XmlProcessorTestUtil {
28      private static final String UTF_8 = "UTF-8";
29      private boolean sortAlfabeticalOnly = false;
30      private boolean keepBlankLines = false;
31      private boolean indentBlankLines = false;
32      private String predefinedSortOrder = "default_0_4_0";
33      private boolean expandEmptyElements = true;
34      private String lineSeparator = "\r\n";

37          return new XmlProcessorTestUtil();

43      public void testInputAndExpected(final String inputFileName, final String expectedFileName) throws Exception {
44          String actual = sortXmlAndReturnResult(inputFileName);
45  
46          final String expected = IOUtils.toString(new FileInputStream(expectedFileName), UTF_8);
47  
48          assertEquals(expected, actual);
49      }
50  
51      public String sortXmlAndReturnResult(String inputFileName) throws Exception {
52          final XmlProcessor xmlProcessor = setup(inputFileName);
53          xmlProcessor.sortXml();
54          final ByteArrayOutputStream sortedXmlOutputStream = xmlProcessor.getSortedXml();
55          return sortedXmlOutputStream.toString(UTF_8);
56      }
57  
58      public void testVerifyXmlIsOrdered(final String inputFileName) throws Exception {
59          final XmlProcessor xmlProcessor = setup(inputFileName);
60          xmlProcessor.sortXml();
61          assertEquals(true, xmlProcessor.isXmlOrdered().isOrdered());
62      }
63  
64      public void testVerifyXmlIsNotOrdered(final String inputFileName, String infoMessage) throws Exception {
65          final XmlProcessor xmlProcessor = setup(inputFileName);
66          xmlProcessor.sortXml();
67          XmlOrderedResult xmlOrdered = xmlProcessor.isXmlOrdered();
68          assertEquals(false, xmlOrdered.isOrdered());
69          assertEquals(infoMessage, xmlOrdered.getErrorMessage());
70      }
71  
72      private XmlProcessor setup(String inputFileName) throws Exception {
73          PluginParameters pluginParameters = new PluginParametersBuilder()
74                  .setPomFile(null)
75                  .setBackupInfo(false, ".bak")
76                  .setEncoding("UTF-8")
77                  .setFormatting(lineSeparator, expandEmptyElements, keepBlankLines)
78                  .setIndent(2, indentBlankLines)
79                  .setSortOrder(predefinedSortOrder + ".xml", null)
80                  .setSortEntities("", "", false).createPluginParameters();
81          final String xml = IOUtils.toString(new FileInputStream(inputFileName), UTF_8);
82  
83          final FileUtil fileUtil = new FileUtil();
84          fileUtil.setup(pluginParameters);
85  
86          WrapperFactory wrapperFactory = new WrapperFactoryImpl(fileUtil);
87          ((WrapperFactoryImpl) wrapperFactory).setup(pluginParameters);
88  
89          final XmlProcessor xmlProcessor = new XmlProcessor(wrapperFactory);
90          xmlProcessor.setup(pluginParameters);
91  
92          if (sortAlfabeticalOnly) {
93              wrapperFactory = new WrapperFactory() {
94  
95                  @Override
96                  public HierarchyWrapper createFromRootElement(final Element rootElement) {
97                      return new HierarchyWrapper(new AlphabeticalSortedWrapper(rootElement));
98                  }
99  
100                 @SuppressWarnings("unchecked")
101                 @Override
102                 public <T extends Content> Wrapper<T> create(final T content) {
103                     if (content instanceof Element) {
104                         Element element = (Element) content;
105                         return (Wrapper<T>) new AlphabeticalSortedWrapper(element);
106                     }
107                     return new UnsortedWrapper<T>(content);
108                 }
109 
110             };
111         } else {
112             new ReflectionHelper(wrapperFactory).setField(fileUtil);
113         }
114         new ReflectionHelper(xmlProcessor).setField(wrapperFactory);
115         xmlProcessor.setOriginalXml(new ByteArrayInputStream(xml.getBytes(UTF_8)));
116         return xmlProcessor;
117     }
