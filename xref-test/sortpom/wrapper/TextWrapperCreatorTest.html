4   import org.junit.Before;

6   import sortpom.parameter.PluginParametersBuilder;

8   import static org.junit.Assert.assertEquals;

10  /**
11   * @author bjorn
12   * @since 2012-06-19
13   */
14  public class TextWrapperCreatorTest {
15      private final TextWrapperCreator textWrapperCreator = new TextWrapperCreator();
16  
17      @Before
18      public void setup() {
19          textWrapperCreator.setup(new PluginParametersBuilder().setEncoding("UTF-8")
20                  .setFormatting("\n", true, true).createPluginParameters());
21      }
22  
23      @Test
24      public void testIsEmptyLine() {
25          assertEquals(false, textWrapperCreator.isBlankLineOrLines(new Text("\n      sortpom\n  ")));
26          assertEquals(false, textWrapperCreator.isBlankLineOrLines(new Text("sortpom")));
27          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("\n  ")));
28          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("  \n  ")));
29          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("  \n\n  ")));
30          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("\n\n")));
31          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("  \r  ")));
32          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("  \r\r  ")));
33          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("\r\r")));
34          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("  \r\n  ")));
35          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("  \r\n\r\n  ")));
36          assertEquals(true, textWrapperCreator.isBlankLineOrLines(new Text("\r\n\r\n")));
37          assertEquals(false, textWrapperCreator.isBlankLineOrLines(new Text("  ")));
38      }
39  }
