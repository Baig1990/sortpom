23  public class ElementToStringTest {
24      private static final String UTF_8 = "UTF-8";
25  
26      @Test
27      public void testToString() throws Exception {
28          String expected = IOUtils.toString(new FileInputStream("src/test/resources/Real1_expected_toString.txt"), UTF_8);
29          assertEquals(expected, getToStringOnRootElementWrapper("Real1_input.xml"));
30      }
31  
32      private String getToStringOnRootElementWrapper(String inputFileName) throws IOException, JDOMException {
33          PluginParameters pluginParameters = new PluginParametersBuilder()
34                  .setPomFile(null).setBackupInfo(false, ".bak")
35                  .setEncoding("UTF-8")
36                  .setFormatting("\r\n", true, true)
37                  .setIndent(2, false)
38                  .setSortOrder("default_0_4_0.xml", null)
39                  .setSortEntities("scope,groupId,artifactId", "groupId,artifactId", true).createPluginParameters();
40  
41          FileUtil fileUtil = new FileUtil();
42          fileUtil.setup(pluginParameters);
43  
44          String xml = IOUtils.toString(new FileInputStream("src/test/resources/" + inputFileName), UTF_8);
45          SAXBuilder parser = new SAXBuilder();
46          Document document = parser.build(new ByteArrayInputStream(xml.getBytes(UTF_8)));
47  
48          WrapperFactoryImpl wrapperFactory = new WrapperFactoryImpl(fileUtil);
49          wrapperFactory.setup(pluginParameters);
50          HierarchyWrapper rootWrapper = wrapperFactory.createFromRootElement(document.getRootElement());
51          rootWrapper.createWrappedStructure(wrapperFactory);
52  
53          return rootWrapper.toString();
54      }
