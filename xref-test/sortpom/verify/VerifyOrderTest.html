11  public class VerifyOrderTest {
12  
13      @Test
14      public final void sortedButUnformattedSortOrderShouldPass() throws Exception {
15          SortPomImplUtil.create()
16                  .predefinedSortOrder("recommended_2008_06")
17                  .testVerifyXmlIsOrdered("/Real2_input.xml");
18      }
19  
20      @Test
21      public final void wrongSortedShouldNotPass() throws Exception {
22          SortPomImplUtil.create()
23                  .predefinedSortOrder("custom_1")
24                  .testVerifyXmlIsNotOrdered("/Real2_input.xml",
25                          "The xml element <properties> should be placed before <inceptionYear>");
26      }
27  
28      @Test
29      public void unsortedDefaultVerifyShouldPerformSort() throws Exception {
30          SortPomImplUtil.create()
31                  .testVerifySort("/Real1_input.xml", "/Real1_expected.xml", "[INFO] The xml element <version> should be placed before <name>");
32      }
33  
34      @Test
35      public void unsortedSortVerifyShouldPerformSort() throws Exception {
36          SortPomImplUtil.create()
37                  .verifyFail("SORT")
38                  .testVerifySort("/Real1_input.xml", "/Real1_expected.xml", "[INFO] The xml element <version> should be placed before <name>");
39      }
40  
41      @Test
42      public void unsortedStopVerifyShouldPerformSort() throws Exception {
43          SortPomImplUtil.create()
44                  .verifyFail("STOP")
45                  .testVerifyFail("/Real1_input.xml", FailureException.class, "[ERROR] The xml element <version> should be placed before <name>");
46      }
47  
48      @Test
49      public void unsortedWarnVerifyShouldPerformSort() throws Exception {
50          SortPomImplUtil.create()
51                  .verifyFail("WARN")
52                  .testVerifyWarn("/Real1_input.xml",
53                          "[WARNING] The xml element <version> should be placed before <name>");
54      }
