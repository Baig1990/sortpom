28      @Test
29      public void multipleErrorsShouldBeReportedInLogger() throws Exception {
30          String xml = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
31                  "<project xmlns=\"http://maven.apache.org/POM/4.0.0\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://maven.apache.org/POM/4.0.0 http://maven.apache.org/maven-v4_0_0.xsd\">\n" +
32                  "  <artifactId>sortpom</artifactId>\n" +
33                  "  <description name=\"pelle\" id=\"id\" other=\"övrigt\">Här använder vi åäö</description>\n" +
34                  "  <groupId>sortpom</groupId>\n" +
35                  "  <modelVersion>4.0.0</modelVersion>\n" +
36                  "  <name>SortPom</name>\n" +
37                  "  <!-- Egenskaper för projektet -->\n" +
38                  "  <properties>\n" +
39                  "    <?sortpom resume?>" +
40                  "<compileSource>1.6</compileSource>\n" +
41                  "    <?sortpom ignore?>" +
42                  "    <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>\n" +
43                  "    <?sortpom resume?>" +
44                  "  </properties>\n" +
45                  "    <?sortpom token='0'?>" +
46                  "    <?sortpom gurka?>" +
47                  "  <reporting />\n" +
48                  "    <?sortpom ignore?>" +
49                  "  <version>1.0.0-SNAPSHOT</version>\n" +
50                  "</project>";
51          try {
52              parser.scanForIgnoredSections(xml);
53              fail();
54          } catch (FailureException fex) {
55              assertThat(fex.getMessage(), is("Xml contained unexpected sortpom instruction 'resume'. Please use expected instruction <?sortpom IGNORE?>"));
56          }
57  
58          verify(logger).error("Xml contained unexpected sortpom instruction 'resume'. Please use expected instruction <?sortpom IGNORE?>");
59          verify(logger).error("Xml contained unknown sortpom instruction 'token='0''. Please use <?sortpom IGNORE?> or <?sortpom RESUME?>");
60          verify(logger).error("Xml contained unknown sortpom instruction 'gurka'. Please use <?sortpom IGNORE?> or <?sortpom RESUME?>");
61          verify(logger).error("Xml processing instructions for sortpom was not properly terminated. Every <?sortpom IGNORE?> must be followed with <?sortpom RESUME?>");
62          verifyNoMoreInteractions(logger);
63      }

65      @Test
66      public void replaceMultipleSectionShouldCreateManyTokens() throws Exception {
67          String xml = "abc<?sortpom ignore?>def0<?sortpom resume?>cbaabc<?SORTPOM Ignore?>def1<?sortPom reSUME?>cba";
68          parser.scanForIgnoredSections(xml);
69          String replaced = parser.replaceIgnoredSections();
70  
71          assertThat(replaced, is("abc<?sortpom token='0'?>cbaabc<?sortpom token='1'?>cba"));
72          assertThat(parser.existsIgnoredSections(), is(true));
73          verifyNoMoreInteractions(logger);
74      }
75  
76      @Test
77      public void revertSectionsInRearrangedOrderShouldPlaceTextInRightOrder() throws Exception {
78          String xml = "abc<?sortpom ignore?>def0<?sortpom resume?>cbaabc<?SORTPOM Ignore?>def1<?sortPom reSUME?>cba";
79          parser.scanForIgnoredSections(xml);
80          String replaced = parser.replaceIgnoredSections();
81          assertThat(replaced, is("abc<?sortpom token='0'?>cbaabc<?sortpom token='1'?>cba"));
82  
83          String sortedXml = "abc<?sortpom token='1'?>cbaabc<?sortpom token='0'?>cba";
84          String outputXml = parser.revertIgnoredSections(sortedXml);
85  
86          assertThat(outputXml, is("abc<?SORTPOM Ignore?>def1<?sortPom reSUME?>cbaabc<?sortpom ignore?>def0<?sortpom resume?>cba"));
87          verifyNoMoreInteractions(logger);
88      }
