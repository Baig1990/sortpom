001package sortpom.processinstruction;
002
003import org.junit.Before;
004import org.junit.Test;
005import sortpom.exception.FailureException;
006import sortpom.logger.SortPomLogger;
007
008import static org.hamcrest.Matchers.is;
009import static org.junit.Assert.assertThat;
010import static org.junit.Assert.fail;
011import static org.mockito.Mockito.*;
012
013/**
014 * @author bjorn
015 * @since 2013-12-28
016 */
017public class XmlProcessingInstructionParserTest {
018
019    private XmlProcessingInstructionParser parser;
020    private SortPomLogger logger = mock(SortPomLogger.class);
021
022    @Before
023    public void setUp() throws Exception {
024        parser = new XmlProcessingInstructionParser();
025        parser.setup(logger);
026    }
027
028    @Test
029    public void multipleErrorsShouldBeReportedInLogger() throws Exception {
030        String xml = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
031                "<project xmlns=\"http://maven.apache.org/POM/4.0.0\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://maven.apache.org/POM/4.0.0 http://maven.apache.org/maven-v4_0_0.xsd\">\n" +
032                "  <artifactId>sortpom</artifactId>\n" +
033                "  <description name=\"pelle\" id=\"id\" other=\"övrigt\">Här använder vi åäö</description>\n" +
034                "  <groupId>sortpom</groupId>\n" +
035                "  <modelVersion>4.0.0</modelVersion>\n" +
036                "  <name>SortPom</name>\n" +
037                "  <!-- Egenskaper för projektet -->\n" +
038                "  <properties>\n" +
039                "    <?sortpom resume?>" +
040                "<compileSource>1.6</compileSource>\n" +
041                "    <?sortpom ignore?>" +
042                "    <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>\n" +
043                "    <?sortpom resume?>" +
044                "  </properties>\n" +
045                "    <?sortpom token='0'?>" +
046                "    <?sortpom gurka?>" +
047                "  <reporting />\n" +
048                "    <?sortpom ignore?>" +
049                "  <version>1.0.0-SNAPSHOT</version>\n" +
050                "</project>";
051        try {
052            parser.scanForIgnoredSections(xml);
053            fail();
054        } catch (FailureException fex) {
055            assertThat(fex.getMessage(), is("Xml contained unexpected sortpom instruction 'resume'. Please use expected instruction <?sortpom IGNORE?>"));
056        }
057
058        verify(logger).error("Xml contained unexpected sortpom instruction 'resume'. Please use expected instruction <?sortpom IGNORE?>");
059        verify(logger).error("Xml contained unknown sortpom instruction 'token='0''. Please use <?sortpom IGNORE?> or <?sortpom RESUME?>");
060        verify(logger).error("Xml contained unknown sortpom instruction 'gurka'. Please use <?sortpom IGNORE?> or <?sortpom RESUME?>");
061        verify(logger).error("Xml processing instructions for sortpom was not properly terminated. Every <?sortpom IGNORE?> must be followed with <?sortpom RESUME?>");
062        verifyNoMoreInteractions(logger);
063    }
064
065    @Test
066    public void replaceMultipleSectionShouldCreateManyTokens() throws Exception {
067        String xml = "abc<?sortpom ignore?>def0<?sortpom resume?>cbaabc<?SORTPOM Ignore?>def1<?sortPom reSUME?>cba";
068        parser.scanForIgnoredSections(xml);
069        String replaced = parser.replaceIgnoredSections();
070
071        assertThat(replaced, is("abc<?sortpom token='0'?>cbaabc<?sortpom token='1'?>cba"));
072        assertThat(parser.existsIgnoredSections(), is(true));
073        verifyNoMoreInteractions(logger);
074    }
075
076    @Test
077    public void revertSectionsInRearrangedOrderShouldPlaceTextInRightOrder() throws Exception {
078        String xml = "abc<?sortpom ignore?>def0<?sortpom resume?>cbaabc<?SORTPOM Ignore?>def1<?sortPom reSUME?>cba";
079        parser.scanForIgnoredSections(xml);
080        String replaced = parser.replaceIgnoredSections();
081        assertThat(replaced, is("abc<?sortpom token='0'?>cbaabc<?sortpom token='1'?>cba"));
082
083        String sortedXml = "abc<?sortpom token='1'?>cbaabc<?sortpom token='0'?>cba";
084        String outputXml = parser.revertIgnoredSections(sortedXml);
085
086        assertThat(outputXml, is("abc<?SORTPOM Ignore?>def1<?sortPom reSUME?>cbaabc<?sortpom ignore?>def0<?sortpom resume?>cba"));
087        verifyNoMoreInteractions(logger);
088    }
089
090    @Test
091    public void noInstructionsShouldWork() throws Exception {
092        String xml = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
093                "<project xmlns=\"http://maven.apache.org/POM/4.0.0\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://maven.apache.org/POM/4.0.0 http://maven.apache.org/maven-v4_0_0.xsd\">\n" +
094                "  <artifactId>sortpom</artifactId>\n" +
095                "  <description name=\"pelle\" id=\"id\" other=\"övrigt\">Här använder vi åäö</description>\n" +
096                "  <groupId>sortpom</groupId>\n" +
097                "  <modelVersion>4.0.0</modelVersion>\n" +
098                "  <name>SortPom</name>\n" +
099                "  <!-- Egenskaper för projektet -->\n" +
100                "  <properties>\n" +
101                "    <compileSource>1.6</compileSource>\n" +
102                "    <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>\n" +
103                "  </properties>\n" +
104                "  <reporting />\n" +
105                "  <version>1.0.0-SNAPSHOT</version>\n" +
106                "</project>";
107        parser.scanForIgnoredSections(xml);
108
109        assertThat(parser.existsIgnoredSections(), is(false));
110
111        String replaced = parser.replaceIgnoredSections();
112        String outputXml = parser.revertIgnoredSections(xml);
113
114        assertThat(replaced, is(xml));
115        assertThat(outputXml, is(xml));
116        verifyNoMoreInteractions(logger);
117    }
118}
