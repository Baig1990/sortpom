021    @Before
022    public void setUp() throws Exception {
023        ignoredSectionsStore = new IgnoredSectionsStore();
024        ignoredSections = new ReflectionHelper(ignoredSectionsStore).getField(ArrayList.class);
025    }

027    @Test
028    public void replaceNoSectionShouldReturnSameXml() throws Exception {
029        String xml = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" +
030                "<project xmlns=\"http://maven.apache.org/POM/4.0.0\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://maven.apache.org/POM/4.0.0 http://maven.apache.org/maven-v4_0_0.xsd\">\n" +
031                "  <artifactId>sortpom</artifactId>\n" +
032                "  <description name=\"pelle\" id=\"id\" other=\"övrigt\">Här använder vi åäö</description>\n" +
033                "  <groupId>sortpom</groupId>\n" +
034                "  <modelVersion>4.0.0</modelVersion>\n" +
035                "  <name>SortPom</name>\n" +
036                "  <!-- Egenskaper för projektet -->\n" +
037                "  <properties>\n" +
038                "    <compileSource>1.6</compileSource>\n" +
039                "    <project.build.sourceEncoding>UTF-8</project.build.sourceEncoding>\n" +
040                "  </properties>\n" +
041                "  <reporting />\n" +
042                "  <version>1.0.0-SNAPSHOT</version>\n" +
043                "</project>";
044        String replaced = ignoredSectionsStore.replaceIgnoredSections(xml);
045
046        assertThat(replaced, is(xml));
047        assertThat(ignoredSections.size(), is(0));
048    }
049
050    @Test
051    public void replaceOneSectionShouldCreateOneToken() throws Exception {
052        String xml = "abc<?sortpom ignore?>def<?sortpom resume?>cba";
053        String replaced = ignoredSectionsStore.replaceIgnoredSections(xml);
054
055        assertThat(replaced, is("abc<?sortpom token='0'?>cba"));
056        assertThat(ignoredSections.size(), is(1));
057        assertThat(ignoredSections.get(0), is("<?sortpom ignore?>def<?sortpom resume?>"));
058    }
059
060    @Test
061    public void replaceMultipleSectionShouldCreateManyTokens() throws Exception {
062        String xml = "abc<?sortpom ignore?>def1<?sortpom resume?>cbaabc<?SORTPOM Ignore?>def2<?sortPom reSUME?>cba";
063        String replaced = ignoredSectionsStore.replaceIgnoredSections(xml);
064
065        assertThat(replaced, is("abc<?sortpom token='0'?>cbaabc<?sortpom token='1'?>cba"));
066        assertThat(ignoredSections.size(), is(2));
067        assertThat(ignoredSections.get(0), is("<?sortpom ignore?>def1<?sortpom resume?>"));
068        assertThat(ignoredSections.get(1), is("<?SORTPOM Ignore?>def2<?sortPom reSUME?>"));
069    }

129    @Test
130    public void revertTokensInOrderShouldWork() throws Exception {
131        String xml = "abc<?sortpom token='0'?>cbaabc<?sortpom token='1'?>cba";
132        ignoredSections.add("<?sortpom ignore?>def1<?sortpom resume?>");
133        ignoredSections.add("<?SORTPOM Ignore?>def2<?sortPom reSUME?>");
134
135        String replaced = ignoredSectionsStore.revertIgnoredSections(xml);
136        assertThat(replaced, is("abc<?sortpom ignore?>def1<?sortpom resume?>cbaabc<?SORTPOM Ignore?>def2<?sortPom reSUME?>cba"));
137    }
138
139    @Test
140    public void revertTokensInRearrangedOrderShouldPlaceTextInRightOrder() throws Exception {
141        String xml = "abc<?sortpom token='1'?>cbaabc<?sortpom token='0'?>cba";
142        ignoredSections.add("<?sortpom ignore?>def0<?sortpom resume?>");
143        ignoredSections.add("<?SORTPOM Ignore?>def1<?sortPom reSUME?>");
144
145        String replaced = ignoredSectionsStore.revertIgnoredSections(xml);
146        assertThat(replaced, is("abc<?SORTPOM Ignore?>def1<?sortPom reSUME?>cbaabc<?sortpom ignore?>def0<?sortpom resume?>cba"));
147    }
