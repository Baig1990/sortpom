001package sortpom.util;
002
003import org.apache.commons.io.IOUtils;
004import org.jdom.Content;
005import org.jdom.Element;
006import refutils.ReflectionHelper;
007import sortpom.XmlProcessor;
008import sortpom.parameter.PluginParameters;
009import sortpom.parameter.PluginParametersBuilder;
010import sortpom.wrapper.*;
011import sortpom.wrapper.content.AlphabeticalSortedWrapper;
012import sortpom.wrapper.content.UnsortedWrapper;
013import sortpom.wrapper.content.Wrapper;
014import sortpom.wrapper.operation.HierarchyWrapper;
015import sortpom.wrapper.operation.WrapperFactory;
016
017import java.io.ByteArrayInputStream;
018import java.io.ByteArrayOutputStream;
019import java.io.FileInputStream;
020
021import static org.junit.Assert.assertEquals;
022
023/**
024 * @author bjorn
025 * @since 2012-06-10
026 */
027public class XmlProcessorTestUtil {
028    private static final String UTF_8 = "UTF-8";
029    private boolean sortAlfabeticalOnly = false;
030    private boolean keepBlankLines = false;
031    private boolean indentBlankLines = false;
032    private String predefinedSortOrder = "default_0_4_0";
033    private boolean expandEmptyElements = true;
034    private String lineSeparator = "\r\n";
035
036    public static XmlProcessorTestUtil create() {
037        return new XmlProcessorTestUtil();
038    }
039
040    private XmlProcessorTestUtil() {
041    }
042
043    public void testInputAndExpected(final String inputFileName, final String expectedFileName) throws Exception {
044        String actual = sortXmlAndReturnResult(inputFileName);
045
046        final String expected = IOUtils.toString(new FileInputStream(expectedFileName), UTF_8);
047
048        assertEquals(expected, actual);
049    }
050
051    public String sortXmlAndReturnResult(String inputFileName) throws Exception {
052        final XmlProcessor xmlProcessor = setup(inputFileName);
053        xmlProcessor.sortXml();
054        final ByteArrayOutputStream sortedXmlOutputStream = xmlProcessor.getSortedXml();
055        return sortedXmlOutputStream.toString(UTF_8);
056    }
057
058    public void testVerifyXmlIsOrdered(final String inputFileName) throws Exception {
059        final XmlProcessor xmlProcessor = setup(inputFileName);
060        xmlProcessor.sortXml();
061        assertEquals(true, xmlProcessor.isXmlOrdered().isOrdered());
062    }
063
064    public void testVerifyXmlIsNotOrdered(final String inputFileName, String infoMessage) throws Exception {
065        final XmlProcessor xmlProcessor = setup(inputFileName);
066        xmlProcessor.sortXml();
067        XmlOrderedResult xmlOrdered = xmlProcessor.isXmlOrdered();
068        assertEquals(false, xmlOrdered.isOrdered());
069        assertEquals(infoMessage, xmlOrdered.getErrorMessage());
070    }
071
072    private XmlProcessor setup(String inputFileName) throws Exception {
073        PluginParameters pluginParameters = new PluginParametersBuilder()
074                .setPomFile(null)
075                .setBackupInfo(false, ".bak")
076                .setEncoding("UTF-8")
077                .setFormatting(lineSeparator, expandEmptyElements, keepBlankLines)
078                .setIndent(2, indentBlankLines)
079                .setSortOrder(predefinedSortOrder + ".xml", null)
080                .setSortEntities("", "", false).createPluginParameters();
081        final String xml = IOUtils.toString(new FileInputStream(inputFileName), UTF_8);
082
083        final FileUtil fileUtil = new FileUtil();
084        fileUtil.setup(pluginParameters);
085
086        WrapperFactory wrapperFactory = new WrapperFactoryImpl(fileUtil);
087        ((WrapperFactoryImpl) wrapperFactory).setup(pluginParameters);
088
089        final XmlProcessor xmlProcessor = new XmlProcessor(wrapperFactory);
090        xmlProcessor.setup(pluginParameters);
091
092        if (sortAlfabeticalOnly) {
093            wrapperFactory = new WrapperFactory() {
094
095                @Override
096                public HierarchyWrapper createFromRootElement(final Element rootElement) {
097                    return new HierarchyWrapper(new AlphabeticalSortedWrapper(rootElement));
098                }
099
100                @SuppressWarnings("unchecked")
101                @Override
102                public <T extends Content> Wrapper<T> create(final T content) {
103                    if (content instanceof Element) {
104                        Element element = (Element) content;
105                        return (Wrapper<T>) new AlphabeticalSortedWrapper(element);
106                    }
107                    return new UnsortedWrapper<T>(content);
108                }
109
110            };
111        } else {
112            new ReflectionHelper(wrapperFactory).setField(fileUtil);
113        }
114        new ReflectionHelper(xmlProcessor).setField(wrapperFactory);
115        xmlProcessor.setOriginalXml(new ByteArrayInputStream(xml.getBytes(UTF_8)));
116        return xmlProcessor;
117    }
118
119    public XmlProcessorTestUtil sortAlfabeticalOnly() {
120        sortAlfabeticalOnly = true;
121        return this;
122    }
123
124    public XmlProcessorTestUtil keepBlankLines() {
125        keepBlankLines = true;
126        return this;
127    }
128
129    public XmlProcessorTestUtil lineSeparator(String lineSeparator) {
130        this.lineSeparator = lineSeparator;
131        return this;
132    }
133
134    public XmlProcessorTestUtil indentBlankLines() {
135        indentBlankLines = true;
136        return this;
137    }
138
139    public XmlProcessorTestUtil predefinedSortOrder(String predefinedSortOrder) {
140        this.predefinedSortOrder = predefinedSortOrder;
141        return this;
142    }
143
144    public XmlProcessorTestUtil expandEmptyElements(boolean expand) {
145        this.expandEmptyElements = expand;
146        return this;
147    }
148}
