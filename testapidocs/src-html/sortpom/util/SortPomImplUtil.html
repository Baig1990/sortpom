001package sortpom.util;
002
003import sortpom.parameter.PluginParameters;
004import sortpom.parameter.PluginParametersBuilder;
005
006import java.io.File;
007import java.util.List;
008
009import static org.hamcrest.Matchers.endsWith;
010import static org.hamcrest.Matchers.startsWith;
011import static org.junit.Assert.*;
012
013public class SortPomImplUtil {
014
015    private TestHandler testHandler;
016
017    private String defaultOrderFileName = "default_0_4_0.xml";
018    private String sortDependencies = "";
019    private String sortPlugins = "";
020    private boolean sortProperties = false;
021    private String predefinedSortOrder = "";
022    private String lineSeparator = "\r\n";
023    private String testPomFileName = "src/test/resources/testpom.xml";
024    private String testPomBackupExtension = ".testExtension";
025
026    private int nrOfIndentSpace = 2;
027    private boolean keepBlankLines = false;
028    private boolean indentBLankLines = false;
029    private String verifyFail = "SORT";
030    private String encoding = TestHandler.UTF_8;
031    private File testpom;
032
033    private SortPomImplUtil() {
034    }
035
036    public static SortPomImplUtil create() {
037        return new SortPomImplUtil();
038    }
039
040    public void testFiles(final String inputResourceFileName, final String expectedResourceFileName)
041            throws Exception {
042        setup();
043        testHandler = new TestHandler(inputResourceFileName, expectedResourceFileName, getPluginParameters());
044        testHandler.performTest();
045    }
046
047    public List<String> testFilesAndReturnLogs(final String inputResourceFileName, final String expectedResourceFileName)
048            throws Exception {
049        setup();
050        testHandler = new TestHandler(inputResourceFileName, expectedResourceFileName, getPluginParameters());
051        testHandler.performTest();
052        return testHandler.getInfoLogger();
053    }
054
055    public void testNoSorting(final String inputResourceFileName)
056            throws Exception {
057        setup();
058        testHandler = new TestHandler(inputResourceFileName, inputResourceFileName, getPluginParameters());
059        testHandler.performNoSortTest();
060        assertEquals("[INFO] Pom file is already sorted, exiting", testHandler.getInfoLogger().get(1));
061    }
062
063    public void testVerifyXmlIsOrdered(final String inputResourceFileName)
064            throws Exception {
065        setup();
066        testHandler = new TestHandler(inputResourceFileName, getPluginParameters());
067        XmlOrderedResult xmlOrderedResult = testHandler.performVerify();
068        assertEquals("Expected that xml is ordered, ", true, xmlOrderedResult.isOrdered());
069    }
070
071    public void testVerifyXmlIsNotOrdered(final String inputResourceFileName, CharSequence warningMessage)
072            throws Exception {
073        setup();
074        testHandler = new TestHandler(inputResourceFileName, getPluginParameters());
075        XmlOrderedResult xmlOrderedResult = testHandler.performVerify();
076        assertEquals("Expected that xml is not ordered, ", false, xmlOrderedResult.isOrdered());
077        assertEquals(warningMessage, xmlOrderedResult.getErrorMessage());
078    }
079
080    public void testVerifySort(final String inputResourceFileName, final String expectedResourceFileName, String warningMessage)
081            throws Exception {
082        setup();
083        testHandler = new TestHandler(inputResourceFileName, expectedResourceFileName, getPluginParameters());
084        testHandler.performTestThatSorted();
085        assertThat(testHandler.getInfoLogger().get(0), startsWith("[INFO] Verifying file "));
086        assertEquals(warningMessage, testHandler.getInfoLogger().get(1));
087        assertThat(testHandler.getInfoLogger().get(2), startsWith("[INFO] The file "));
088        assertThat(testHandler.getInfoLogger().get(2), endsWith(" is not sorted"));
089        assertThat(testHandler.getInfoLogger().get(3), startsWith("[INFO] Sorting file "));
090    }
091
092    public void testVerifyFail(String inputResourceFileName, Class<?> expectedExceptionClass, String warningMessage) {
093        setup();
094        testHandler = new TestHandler(inputResourceFileName, getPluginParameters());
095        try {
096            testHandler.performTestThatDidNotSort();
097            fail();
098        } catch (Exception e) {
099            assertEquals(expectedExceptionClass, e.getClass());
100            assertThat(testHandler.getInfoLogger().get(0), startsWith("[INFO] Verifying file "));
101            assertEquals(warningMessage, testHandler.getInfoLogger().get(1));
102            assertThat(testHandler.getInfoLogger().get(2), startsWith("[ERROR] The file "));
103            assertThat(testHandler.getInfoLogger().get(2), endsWith(" is not sorted"));
104        }
105    }
106
107    public void testVerifyWarn(String inputResourceFileName, String warningMessage) throws Exception {
108        setup();
109        testHandler = new TestHandler(inputResourceFileName, inputResourceFileName, getPluginParameters());
110        testHandler.performTestThatDidNotSort();
111        assertThat(testHandler.getInfoLogger().get(0), startsWith("[INFO] Verifying file "));
112        assertEquals(warningMessage, testHandler.getInfoLogger().get(1));
113        assertThat(testHandler.getInfoLogger().get(2), startsWith("[WARNING] The file "));
114        assertThat(testHandler.getInfoLogger().get(2), endsWith(" is not sorted"));
115    }
116
117    public SortPomImplUtil nrOfIndentSpace(int indent) {
118        nrOfIndentSpace = indent;
119        return this;
120    }
121
122    public SortPomImplUtil keepBlankLines() {
123        keepBlankLines = true;
124        return this;
125    }
126
127    public SortPomImplUtil indentBLankLines() {
128        indentBLankLines = true;
129        return this;
130    }
131
132    public SortPomImplUtil sortDependencies(String sortOrder) {
133        sortDependencies = sortOrder;
134        return this;
135    }
136
137    public SortPomImplUtil sortPlugins(String sortOrder) {
138        sortPlugins = sortOrder;
139        return this;
140    }
141
142    public SortPomImplUtil sortProperties() {
143        sortProperties = true;
144        return this;
145    }
146
147    public SortPomImplUtil defaultOrderFileName(String defaultOrderFileName) {
148        this.defaultOrderFileName = defaultOrderFileName;
149        return this;
150    }
151
152    public SortPomImplUtil predefinedSortOrder(String predefinedSortOrder) {
153        this.predefinedSortOrder = predefinedSortOrder;
154        this.defaultOrderFileName = null;
155        return this;
156    }
157
158    public SortPomImplUtil lineSeparator(String lineSeparator) {
159        this.lineSeparator = lineSeparator;
160        return this;
161    }
162
163    public SortPomImplUtil verifyFail(String verifyFail) {
164        this.verifyFail = verifyFail;
165        return this;
166    }
167
168    public SortPomImplUtil backupFileExtension(String backupFileExtension) {
169        this.testPomBackupExtension = backupFileExtension;
170        return this;
171    }
172
173    public SortPomImplUtil encoding(String encoding) {
174        this.encoding = encoding;
175        return this;
176    }
177
178    public SortPomImplUtil testPomFileNameUniqueNumber(int uniqueNumber) {
179        this.testPomFileName = "src/test/resources/testpom" +
180                uniqueNumber + ".xml";
181        return this;
182    }
183
184    private void setup() {
185        testpom = new File(testPomFileName);
186    }
187
188    private PluginParameters getPluginParameters() {
189        return new PluginParametersBuilder()
190                .setPomFile(testpom)
191                .setBackupInfo(true, testPomBackupExtension)
192                .setEncoding(encoding)
193                .setFormatting(lineSeparator,
194                        true, keepBlankLines)
195                .setIndent(nrOfIndentSpace, indentBLankLines)
196                .setSortEntities(sortDependencies, sortPlugins, sortProperties)
197                .setSortOrder(defaultOrderFileName, predefinedSortOrder)
198                .setVerifyFail(verifyFail)
199                .createPluginParameters();
200    }

202}
