001package sortpom.util;
002
003import org.junit.Before;
004import org.junit.Ignore;
005import org.junit.Rule;
006import org.junit.Test;
007import org.junit.rules.ExpectedException;
008import refutils.ReflectionHelper;
009import sortpom.exception.FailureException;
010
011import java.io.File;
012
013import static org.mockito.Mockito.*;
014
015public class FileUtilExceptionsTest {
016    @Rule
017    public ExpectedException thrown = ExpectedException.none();
018
019    private File backupFileMock = mock(File.class);
020
021    private File pomFileMock = mock(File.class);
022
023    private FileUtil originalFileUtil;
024
025    @Before
026    public void setup() {
027        originalFileUtil = new FileUtil();
028    }
029
030    @Test
031    public void whenOldBackupFileCannotBeDeletedAnExceptionShouldBeThrown() throws Exception {
032        FileUtil fileUtil = createFileUtil();
033        doNotAccessRealBackupFile(fileUtil);
034
035        when(backupFileMock.exists()).thenReturn(true);
036        when(backupFileMock.delete()).thenReturn(false);
037
038        thrown.expect(FailureException.class);
039        thrown.expectMessage("Could not remove old backup file, filename: backupFileName");
040
041        fileUtil.backupFile();
042    }
043
044    @Ignore("This test does not work under JDK7 due to internal JDK changes")
045    @Test
046    public void whenSourceFileCannotBeCopiedAnExceptionShouldBeThrown() throws Exception {
047        FileUtil fileUtil = createFileUtil();
048        doNotAccessRealBackupFile(fileUtil);
049
050        when(backupFileMock.exists()).thenReturn(true);
051        when(backupFileMock.delete()).thenReturn(true);
052        when(pomFileMock.getPath()).thenReturn("gurka");
053
054        thrown.expect(FailureException.class);
055        thrown.expectMessage("Could not create backup file to filename: backupFileName");
056
057        fileUtil.backupFile();
058    }
059
060    @Ignore("This test does not work under JDK7 due to internal JDK changes")
061    @Test
062    public void whenPomFileCannotBeReadAnExceptionShouldBeThrown() throws Exception {
063        FileUtil fileUtil = createFileUtil();
064        doNotAccessRealBackupFile(fileUtil);
065
066        when(pomFileMock.getPath()).thenReturn("gurka");
067        when(pomFileMock.getAbsolutePath()).thenReturn("pomFileName");
068
069        thrown.expect(FailureException.class);
070        thrown.expectMessage("Could not read pom file: pomFileName");
071
072        fileUtil.getPomFileContent();
073    }
074
075    @Ignore("This test does not work under JDK7 due to internal JDK changes")
076    @Test
077    public void whenPomFileCannotBeSavedAnExceptionShouldBeThrown() throws Exception {
078        FileUtil fileUtil = createFileUtil();
079        doNotAccessRealBackupFile(fileUtil);
080
081        when(pomFileMock.getPath()).thenReturn("/\\");
082        when(pomFileMock.getAbsolutePath()).thenReturn("pomFileName");
083
084        thrown.expect(FailureException.class);
085        thrown.expectMessage("Could not save sorted pom file: pomFileName");
086
087        fileUtil.savePomFile(null);
088    }
089
090    private FileUtil createFileUtil() throws NoSuchFieldException, IllegalAccessException {
091        ReflectionHelper helper = new ReflectionHelper(originalFileUtil);
092        helper.setField("backupFile", backupFileMock);
093        helper.setField("pomFile", pomFileMock);
094        helper.setField("newName", "backupFileName");
095        return spy(originalFileUtil);
096    }
097
098    private void doNotAccessRealBackupFile(FileUtil fileUtil) {
099        doNothing().when(fileUtil).createFileHandle();
100    }
101
102}
