001package sortpom.parameter;
002
003import org.junit.Rule;
004import org.junit.Test;
005import org.junit.rules.ExpectedException;
006import sortpom.exception.FailureException;
007
008import static org.junit.Assert.assertEquals;
009
010public class IndentCharactersParameterTest {
011    @Rule
012    public ExpectedException thrown = ExpectedException.none();
013
014    @Test
015    public void zeroIndentCharactersShouldResultInEmptyIndentString() {
016        PluginParameters pluginParameters = new PluginParametersBuilder()
017                .setIndent(0, true)
018                .createPluginParameters();
019
020        assertEquals("", pluginParameters.indentCharacters);
021    }
022
023    @Test
024    public void oneIndentCharacterShouldResultInOneSpace() {
025        PluginParameters pluginParameters = new PluginParametersBuilder()
026                .setIndent(1, true)
027                .createPluginParameters();
028
029        assertEquals(" ", pluginParameters.indentCharacters);
030    }
031
032    @Test
033    public void test255IndentCharacterShouldResultIn255Space() {
034        PluginParameters pluginParameters = new PluginParametersBuilder()
035                .setIndent(255, true)
036                .createPluginParameters();
037
038        // Test for only space
039        assertEquals(true, pluginParameters.indentCharacters.matches("^ *$"));
040        assertEquals(255, pluginParameters.indentCharacters.length());
041    }
042
043    @Test
044    public void minusOneIndentCharacterShouldResultInOneTab() {
045        PluginParameters pluginParameters = new PluginParametersBuilder()
046                .setIndent(-1, true)
047                .createPluginParameters();
048
049        assertEquals("\t", pluginParameters.indentCharacters);
050    }
051
052    @Test
053    public void minusTwoShouldFail() {
054        thrown.expect(FailureException.class);
055        thrown.expectMessage("");
056
057        new PluginParametersBuilder()
058                .setIndent(-2, true)
059                .createPluginParameters();
060    }
061
062    @Test
063    public void moreThan255ShouldFail() {
064        thrown.expect(FailureException.class);
065        thrown.expectMessage("");
066
067        new PluginParametersBuilder()
068                .setIndent(256, true)
069                .createPluginParameters();
070    }
071
072}
