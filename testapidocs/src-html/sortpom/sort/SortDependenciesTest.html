001package sortpom.sort;
002
003import org.junit.Test;
004import sortpom.util.SortPomImplUtil;
005
006import java.util.List;
007
008import static org.hamcrest.Matchers.is;
009import static org.junit.Assert.assertThat;
010
011public class SortDependenciesTest {
012
013    @Test
014    public final void scopeInSortDependenciesShouldSortByScope() throws Exception {
015        SortPomImplUtil.create()
016                .defaultOrderFileName("custom_1.xml")
017                .sortDependencies("scope,groupId,artifactId")
018                .lineSeparator("\r\n")
019                .testFiles("/SortDep_input_simpleWithScope.xml", "/SortDep_expected_simpleWithScope2.xml");
020    }
021
022    @Test
023    public final void defaultGroupIdForPluginsShouldWork() throws Exception {
024        SortPomImplUtil.create()
025                .defaultOrderFileName("custom_1.xml")
026                .sortPlugins("groupId,artifactId")
027                .lineSeparator("\n")
028                .nrOfIndentSpace(4)
029                .testFiles("/PluginDefaultName_input.xml", "/PluginDefaultName_expect.xml");
030    }
031
032    @Test
033    public final void deprecatedSortPluginsTrueMessageShouldWork() throws Exception {
034        List<String> logs = SortPomImplUtil.create()
035                .defaultOrderFileName("custom_1.xml")
036                .sortPlugins("true")
037                .lineSeparator("\n")
038                .nrOfIndentSpace(4)
039                .testFilesAndReturnLogs("/PluginDefaultName_input.xml", "/PluginDefaultName_expect.xml");
040
041        assertThat(logs.get(0), is("[WARNING] [DEPRECATED] The 'true' value in sortPlugins is not used anymore, please use value 'groupId,artifactId' instead. In the next major version 'true' or 'false' will cause an error!"));
042    }
043
044    @Test
045    public final void deprecatedSortPluginsFalseMessageShouldWork() throws Exception {
046        List<String> logs = SortPomImplUtil.create()
047                .sortPlugins("false")
048                .testFilesAndReturnLogs("/full_unsorted_input.xml", "/full_expected.xml");
049
050        assertThat(logs.get(0), is("[WARNING] [DEPRECATED] The 'false' value in sortPlugins is not used anymore, please use empty value '' or omit sortPlugins instead. In the next major version 'true' or 'false' will cause an error!"));
051    }
052
053    @Test
054    public final void deprecatedSortDependenciesTrueMessageShouldWork() throws Exception {
055        List<String> logs = SortPomImplUtil.create()
056                .sortDependencies("true")
057                .sortPlugins("true")
058                .testFilesAndReturnLogs("/Simple_input.xml", "/Simple_expected_sortDep.xml");
059
060
061        assertThat(logs.get(0), is("[WARNING] [DEPRECATED] The 'true' value in sortDependencies is not used anymore, please use value 'groupId,artifactId' instead. In the next major version 'true' or 'false' will cause an error!"));
062    }
063
064    @Test
065    public final void deprecatedSortDependenciesFalseMessageShouldWork() throws Exception {
066        List<String> logs = SortPomImplUtil.create()
067                .sortDependencies("false")
068                .testFilesAndReturnLogs("/full_unsorted_input.xml", "/full_expected.xml");
069
070        assertThat(logs.get(0), is("[WARNING] [DEPRECATED] The 'false' value in sortDependencies is not used anymore, please use empty value '' or omit sortDependencies instead. In the next major version 'true' or 'false' will cause an error!"));
071    }
072}
