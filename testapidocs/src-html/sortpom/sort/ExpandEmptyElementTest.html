001package sortpom.sort;
002
003import org.jdom.Document;
004import org.jdom.Element;
005import org.junit.Assert;
006import org.junit.Test;
007import refutils.ReflectionHelper;
008import sortpom.XmlProcessor;
009import sortpom.parameter.PluginParametersBuilder;
010
011public class ExpandEmptyElementTest {
012    @Test
013    public void trueExpandedParameterShouldExpandEmptyXmlElements() throws Exception {
014        XmlProcessor xmlProcessor = new XmlProcessor(null);
015        xmlProcessor.setup(new PluginParametersBuilder()
016                .setEncoding("UTF-8")
017                .setFormatting("\n", true, false)
018                .setIndent(2, false)
019                .createPluginParameters());
020        new ReflectionHelper(xmlProcessor).setField("newDocument", createXmlFragment());
021
022        String actual = xmlProcessor.getSortedXml().toString("UTF-8");
023        Assert.assertEquals("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<Gurka></Gurka>\n", actual);
024    }
025
026    @Test
027    public void falseExpandedParameterShouldCompressEmptyXmlElements() throws Exception {
028        XmlProcessor xmlProcessor = new XmlProcessor(null);
029        xmlProcessor.setup(new PluginParametersBuilder()
030                .setEncoding("UTF-8")
031                .setFormatting("\n", false, false)
032                .setIndent(2, false)
033                .createPluginParameters());
034        new ReflectionHelper(xmlProcessor).setField("newDocument", createXmlFragment());
035
036        String actual = xmlProcessor.getSortedXml().toString("UTF-8");
037        Assert.assertEquals("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<Gurka />\n", actual);
038    }
039
040    private Document createXmlFragment() {
041        Document newDocument = new Document();
042        newDocument.setRootElement(new Element("Gurka"));
043        return newDocument;
044    }
045}
