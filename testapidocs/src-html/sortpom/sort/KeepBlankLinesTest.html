015                .testInputAndExpected("src/test/resources/EmptyRow_input.xml", "src/test/resources/EmptyRow_expected.xml");
016    }

023    }

027        XmlProcessorTestUtil.create()
028                .keepBlankLines()
029                .indentBlankLines()
030                .testInputAndExpected("src/test/resources/Real1_input.xml", "src/test/resources/Real1_expected_keepBlankLines_indented.xml");
031    }

048    @Test
049    public final void simpleLineBreaksShouldNotBePreserved() throws Exception {
050        XmlProcessorTestUtil.create()
051                .keepBlankLines()
052                .testInputAndExpected("src/test/resources/LineBreak_input.xml", "src/test/resources/Character_expected.xml");
053    }
054
055    @Test
056    public final void allLineBreaksInXmlShouldBeNewlines() throws Exception {
057        String actual = XmlProcessorTestUtil.create()
058                .keepBlankLines()
059                .lineSeparator("\n")
060                .sortXmlAndReturnResult("src/test/resources/LineBreak_N_input.xml");
061        
062        assertThat(actual, containsString("</groupId>\n  <artifactId>whitespace-test</artifactId>\n  <version>"));
063        assertThat(actual, containsString(",\nembedded,\nand"));
064    }
065
066    
067    @Test
068    public final void allLineBreaksInXmlShouldBeCarriageReturnNewlines() throws Exception {
069        String actual = XmlProcessorTestUtil.create()
070                .keepBlankLines()
071                .lineSeparator("\r\n")
072                .sortXmlAndReturnResult("src/test/resources/LineBreak_N_input.xml");
073        
074        assertThat(actual, containsString("</groupId>\r\n  <artifactId>whitespace-test</artifactId>\r\n  <version>"));
075        assertThat(actual, containsString(",\r\nembedded,\r\nand"));
076    }
077
078    @Test
079    public final void allLineBreaksInXmlShouldBeCarriageReturn() throws Exception {
080        String actual = XmlProcessorTestUtil.create()
081                .keepBlankLines()
082                .lineSeparator("\r")
083                .sortXmlAndReturnResult("src/test/resources/LineBreak_N_input.xml");
084        
085        assertThat(actual, containsString("</groupId>\r  <artifactId>whitespace-test</artifactId>\r  <version>"));
086        assertThat(actual, containsString(",\rembedded,\rand"));
087    }
