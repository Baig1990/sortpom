001package sortpom.verify;
002
003import org.junit.Test;
004import sortpom.exception.FailureException;
005import sortpom.util.SortPomImplUtil;
006
007/**
008 * @author bjorn
009 * @since 2012-07-01
010 */
011public class VerifyOrderTest {
012
013    @Test
014    public final void sortedButUnformattedSortOrderShouldPass() throws Exception {
015        SortPomImplUtil.create()
016                .predefinedSortOrder("recommended_2008_06")
017                .testVerifyXmlIsOrdered("/Real2_input.xml");
018    }
019
020    @Test
021    public final void wrongSortedShouldNotPass() throws Exception {
022        SortPomImplUtil.create()
023                .predefinedSortOrder("custom_1")
024                .testVerifyXmlIsNotOrdered("/Real2_input.xml",
025                        "The xml element <properties> should be placed before <inceptionYear>");
026    }
027
028    @Test
029    public void unsortedDefaultVerifyShouldPerformSort() throws Exception {
030        SortPomImplUtil.create()
031                .testVerifySort("/Real1_input.xml", "/Real1_expected.xml", "[INFO] The xml element <version> should be placed before <name>");
032    }
033
034    @Test
035    public void unsortedSortVerifyShouldPerformSort() throws Exception {
036        SortPomImplUtil.create()
037                .verifyFail("SORT")
038                .testVerifySort("/Real1_input.xml", "/Real1_expected.xml", "[INFO] The xml element <version> should be placed before <name>");
039    }
040
041    @Test
042    public void unsortedStopVerifyShouldPerformSort() throws Exception {
043        SortPomImplUtil.create()
044                .verifyFail("STOP")
045                .testVerifyFail("/Real1_input.xml", FailureException.class, "[ERROR] The xml element <version> should be placed before <name>");
046    }
047
048    @Test
049    public void unsortedWarnVerifyShouldPerformSort() throws Exception {
050        SortPomImplUtil.create()
051                .verifyFail("WARN")
052                .testVerifyWarn("/Real1_input.xml",
053                        "[WARNING] The xml element <version> should be placed before <name>");
054    }
055
056}
