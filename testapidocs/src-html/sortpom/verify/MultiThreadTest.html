001package sortpom.verify;
002
003import org.junit.Test;
004import sortpom.util.SortPomImplUtil;
005
006import java.util.ArrayList;
007import java.util.List;
008import java.util.concurrent.*;
009import java.util.concurrent.atomic.AtomicInteger;
010
011import static org.junit.Assert.assertEquals;
012
013public class MultiThreadTest {
014    private final AtomicInteger counter = new AtomicInteger(1);
015
016    @Test
017    public final void multipleSortingsShouldNotInterfereWithEachOther() throws InterruptedException, ExecutionException {
018        ExecutorService executorService = new ScheduledThreadPoolExecutor(10);
019        for (int a = 0; a < 10; a++) {
020            testOneConcurrentLoop(executorService);
021        }
022    }
023
024    private void testOneConcurrentLoop(final ExecutorService executorService) throws InterruptedException, ExecutionException {
025        List<Callable<Boolean>> testThreads = getTestThreads();
026        List<Future<Boolean>> futures = executorService.invokeAll(testThreads);
027        assertAllThreadsReturnedTrue(futures);
028    }
029
030    private List<Callable<Boolean>> getTestThreads() {
031        List<Callable<Boolean>> list = new ArrayList<Callable<Boolean>>();
032        list.add(new TestThread("/full_unsorted_input.xml",
033                "/sortOrderFiles/sorted_default_0_4_0.xml", "default_0_4_0"));
034        list.add(new TestThread("/full_unsorted_input.xml",
035                "/sortOrderFiles/sorted_custom_1.xml", "custom_1"));
036        list.add(new TestThread("/full_unsorted_input.xml",
037                "/sortOrderFiles/sorted_recommended_2008_06.xml", "recommended_2008_06"));
038        list.add(new TestThread("/full_unsorted_input.xml",
039                "/sortOrderFiles/sorted_default_1_0_0.xml", "default_1_0_0"));
040        return list;
041    }
042
043    private void assertAllThreadsReturnedTrue(final List<Future<Boolean>> futures) throws InterruptedException, ExecutionException {
044        for (Future<Boolean> future : futures) {
045            assertEquals(true, future.get());
046        }
047    }
048
049    private class TestThread implements Callable<Boolean> {
050        private final String inputResourceFileName;
051        private final String expectedResourceFileName;
052        private final String predefinedSortOrder;
053
054        public TestThread(final String inputResourceFileName,
055                          final String expectedResourceFileName,
056                          final String predefinedSortOrder) {
057            this.inputResourceFileName = inputResourceFileName;
058            this.expectedResourceFileName = expectedResourceFileName;
059            this.predefinedSortOrder = predefinedSortOrder;
060        }
061
062        @Override
063        public Boolean call() {
064            try {
065                SortPomImplUtil.create()
066                        .lineSeparator("\n")
067                        .testPomFileNameUniqueNumber(counter.getAndIncrement())
068                        .predefinedSortOrder(predefinedSortOrder)
069                        .testVerifyXmlIsOrdered(expectedResourceFileName);
070                SortPomImplUtil.create()
071                        .lineSeparator("\n")
072                        .testPomFileNameUniqueNumber(counter.getAndIncrement())
073                        .predefinedSortOrder(predefinedSortOrder)
074                        .testVerifyXmlIsNotOrdered(inputResourceFileName,
075                                "The xml element <modelVersion> should be placed before <parent>");
076            } catch (Exception e) {
077                e.printStackTrace();
078                return false;
079            }
080            return true;  //To change body of implemented methods use File | Settings | File Templates.
081        }
082    }
083}
