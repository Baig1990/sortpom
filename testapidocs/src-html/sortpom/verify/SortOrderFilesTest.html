001package sortpom.verify;
002
003import org.junit.Test;
004import sortpom.util.SortPomImplUtil;
005
006public class SortOrderFilesTest {
007    @Test
008    public final void sortedCustomSortOrderShouldNotTriggerVerify() throws Exception {
009        SortPomImplUtil.create()
010                .defaultOrderFileName("difforder/differentOrder.xml")
011                .lineSeparator("\n")
012                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_differentOrder.xml");
013    }
014
015    @Test
016    public final void unsortedCustomSortOrderShouldTriggerVerify() throws Exception {
017        SortPomImplUtil.create()
018                .defaultOrderFileName("difforder/differentOrder.xml")
019                .lineSeparator("\n")
020                .testVerifyXmlIsNotOrdered("/full_unsorted_input.xml",
021                        "The xml element <modelVersion> should be placed before <parent>");
022    }
023
024    @Test
025    public final void default040ShouldWorkAsPredefinedSortOrder() throws Exception {
026        SortPomImplUtil.create()
027                .predefinedSortOrder("default_0_4_0")
028                .lineSeparator("\n")
029                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_default_0_4_0.xml");
030    }
031
032    @Test
033    public final void custom1ShouldWorkAsPredefinedSortOrder() throws Exception {
034        SortPomImplUtil.create()
035                .predefinedSortOrder("custom_1")
036                .lineSeparator("\n")
037                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_custom_1.xml");
038    }
039
040    @Test
041    public final void recommended2008_06ShouldWorkAsPredefinedSortOrder() throws Exception {
042        SortPomImplUtil.create()
043                .predefinedSortOrder("recommended_2008_06")
044                .lineSeparator("\n")
045                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_recommended_2008_06.xml");
046    }
047
048    @Test
049    public final void default100ShouldWorkAsPredefinedSortOrder() throws Exception {
050        SortPomImplUtil.create()
051                .predefinedSortOrder("default_1_0_0")
052                .lineSeparator("\n")
053                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_default_1_0_0.xml");
054    }
055
056    @Test
057    public final void defaultPredefinedSortOrderShouldWork() throws Exception {
058        SortPomImplUtil.create()
059                .predefinedSortOrder(null)
060                .lineSeparator("\n")
061                .testVerifyXmlIsOrdered("/sortOrderFiles/sorted_default_1_0_0.xml");
062    }
063
064    @Test
065    public void xmlDeviationsShouldNotHarmPlugin() throws Exception {
066        SortPomImplUtil.create()
067                .lineSeparator("\n")
068                .testVerifyXmlIsOrdered("/Xml_deviations_output.xml");
069    }
