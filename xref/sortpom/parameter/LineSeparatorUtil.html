3   import sortpom.exception.FailureException;
4   
5   import java.nio.charset.Charset;
6   import java.util.Arrays;
7   
8   /**
9    * Encapsulates LineSeparation logic.
10   *
11   * @author bjorn
12   */
13  public class LineSeparatorUtil {
14      private final String string;
15  
16      /**
17       * Creates a line separator and makes sure that it is either &#92;n, &#92;r or &#92;r&#92;n
18       *
19       * @param lineSeparatorString The line separator characters
20       */
21      LineSeparatorUtil(final String lineSeparatorString) {
22          string = lineSeparatorString.replaceAll("\\\\r", "\r").replaceAll("\\\\n", "\n");
23          if (isIllegalString()) {
24              throw new FailureException(
25                      "LineSeparator must be either \\n, \\r or \\r\\n, but separator characters were "
26                              + Arrays.toString(lineSeparatorString.getBytes(Charset.defaultCharset())));
27          }
28      }
29  
30      private boolean isIllegalString() {
31          return !("\n".equalsIgnoreCase(string) ||
32                  "\r".equalsIgnoreCase(string) ||
33                  "\r\n".equalsIgnoreCase(string));
34      }
35  
36      @Override
37      public String toString() {
38          return string;
39      }
40  
41  }
