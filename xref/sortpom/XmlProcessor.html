11  import sortpom.jdomcontent.NewlineText;

13  import sortpom.parameter.PluginParameters;

15  import sortpom.util.XmlOrderedResult;

17  import sortpom.wrapper.operation.HierarchyWrapper;
18  import sortpom.wrapper.operation.WrapperFactory;
19  
20  import java.io.ByteArrayOutputStream;
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.Writer;
24  
25  /**
26   * Creates xml structure and sorts it.
27   *
28   * @author Bjorn Ekryd
29   */
30  public class XmlProcessor {
31      private final WrapperFactory factory;
32  
33      private Document originalDocument;
34      private Document newDocument;
35      private String encoding;
36      private LineSeparatorUtil lineSeparatorUtil;
37      private String indentCharacters;
38      private boolean expandEmptyElements;
39      private boolean indentBlankLines;
40  
41      public XmlProcessor(WrapperFactory factory) {
42          this.factory = factory;
43      }
44  
45      /** Setup default configuration */
46      public void setup(PluginParameters pluginParameters) {
47          this.indentCharacters = pluginParameters.indentCharacters;
48          this.lineSeparatorUtil = pluginParameters.lineSeparatorUtil;
49          this.encoding = pluginParameters.encoding;
50          this.expandEmptyElements = pluginParameters.expandEmptyElements;
51          this.indentBlankLines = pluginParameters.indentBlankLines;
52      }
53  
54      /**
55       * Sets the original xml that should be sorted. Builds a dom document of the
56       * xml.
57       *
58       * @param originalXml the new original xml
59       * @throws org.jdom.JDOMException the jDOM exception
60       * @throws java.io.IOException    Signals that an I/O exception has occurred.
61       */
62      public void setOriginalXml(final InputStream originalXml) throws JDOMException, IOException {
63          SAXBuilder parser = new SAXBuilder();
64          originalDocument = parser.build(originalXml);
65      }
66  
67      /** Creates a new dom document that contains the sorted xml. */
68      public void sortXml() {
69          newDocument = (Document) originalDocument.clone();
70          final Element rootElement = (Element) originalDocument.getRootElement().clone();
71  
72          HierarchyWrapper rootWrapper = factory.createFromRootElement(rootElement);
73  
74          rootWrapper.createWrappedStructure(factory);
75          rootWrapper.detachStructure();
76          rootWrapper.sortStructureAttributes();
77          rootWrapper.sortStructureElements();
78          rootWrapper.connectXmlStructure();
79  
80          newDocument.setRootElement(rootWrapper.getElementContent().getContent());
81      }
82  
83      public XmlOrderedResult isXmlOrdered() {
84          ElementComparator elementComparator = new ElementComparator(originalDocument.getRootElement(), newDocument.getRootElement());
85          return elementComparator.isElementOrdered();
86      }
87  
88      /**
89       * Returns the sorted xml as an OutputStream.
90       *
91       * @return the sorted xml
92       * @throws java.io.IOException
93       */
94      public ByteArrayOutputStream getSortedXml() throws IOException {
95          ByteArrayOutputStream sortedXml = new ByteArrayOutputStream();
96          BufferedLineSeparatorOutputStream bufferedLineOutputStream =
97                  new BufferedLineSeparatorOutputStream(lineSeparatorUtil.toString(), sortedXml);
98  
99          XMLOutputter xmlOutputter = new PatchedXMLOutputter(bufferedLineOutputStream, indentBlankLines);
100         xmlOutputter.setFormat(createPrettyFormat());
101         xmlOutputter.output(newDocument, bufferedLineOutputStream);
102 
103         IOUtils.closeQuietly(bufferedLineOutputStream);
104         return sortedXml;
105     }
106 
107     private Format createPrettyFormat() {
108         final Format prettyFormat = Format.getPrettyFormat();
109         prettyFormat.setExpandEmptyElements(expandEmptyElements);
110         prettyFormat.setEncoding(encoding);
111         prettyFormat.setLineSeparator("\n");
112         prettyFormat.setIndent(indentCharacters);
113         return prettyFormat;
114     }
115 
116     private static class PatchedXMLOutputter extends XMLOutputter {
117         private final BufferedLineSeparatorOutputStream bufferedLineOutputStream;
118         private final boolean indentBlankLines;
119 
120         public PatchedXMLOutputter(BufferedLineSeparatorOutputStream bufferedLineOutputStream, boolean indentBlankLines) {
121             this.bufferedLineOutputStream = bufferedLineOutputStream;
122             this.indentBlankLines = indentBlankLines;
123             XMLOutputter.preserveFormat.setLineSeparator("\n");
124         }
125 
126         /** Stop XMLOutputter from printing comment <!-- --> chars if it is just a newline */
127         @Override
128         protected void printComment(Writer stringWriter, Comment comment) throws IOException {
129             if (comment instanceof NewlineText) {
130                 if (!indentBlankLines) {
131                     clearIndentationForCurrentLine(stringWriter);
132                 }
133             } else {
134                 super.printComment(stringWriter, comment);
135             }
136         }
137 
138         private void clearIndentationForCurrentLine(Writer stringWriter) throws IOException {
139             // Force all xml lines to be written to stream (via the writer)
140             stringWriter.flush();
141 
142             // Remove all inset that has just been written since last newline
143             bufferedLineOutputStream.clearLineBuffer();
144         }
145     }
146 }
