5   /**
6    * The NewlineText is not really a special case of comment. Its just that comments
7    * are not subjected to trimming by jdom. The special handling of NewlineText is done
8    * in XmlProcessor.PatchedXMLOutputter
9    *
10   * @author bjorn
11   * @since 2012-05-17
12   */
13  public class NewlineText extends Comment {
14      private static final long serialVersionUID = -7552189498553321263L;
15  
16      /**
17       * This returns a <code>String</code> representation of the
18       * <code>NewlineText</code>, suitable for debugging. If the XML
19       * representation of the <code>Comment</code> is desired,
20       * {@link org.jdom.output.XMLOutputter#outputString(org.jdom.Comment)}
21       * should be used.
22       *
23       * @return <code>String</code> - information about the
24       * <code>Attribute</code>
25       */
26      public String toString() {
27          return "[NewLine]";
28      }
29  
30  }
