3   import java.util.regex.Pattern;

5   /**
6    * Contains the different processing instruction commands that sortpom supports
7    *
8    * @author bjorn
9    * @since 2013-12-28
10   */
11  enum InstructionType {
12      IGNORE, RESUME;
13  
14      /** non-xml compliant pattern, see http://www.cs.sfu.ca/~cameron/REX.html#IV.2 */
15      static final Pattern INSTRUCTION_PATTERN = Pattern.compile("(?i)<\\?sortpom\\s+([\\w\"'*= ]*)\\s*\\?>");
16      static final Pattern IGNORE_SECTIONS_PATTERN = Pattern.compile(
17              "(?is)<\\?sortpom\\s+" + IGNORE + "\\s*\\?>.*?<\\?sortpom\\s+" + RESUME + "\\s*\\?>");
18      static final Pattern TOKEN_PATTERN = Pattern.compile("(?i)<\\?sortpom\\s+token='(\\d+)'\\s*\\?>");
19  
20      public InstructionType next() {
21          if (this == IGNORE) {
22              return RESUME;
23          }
24          return IGNORE;
25      }
26  
27      public static boolean containsType(String instruction) {
28          return IGNORE.name().equalsIgnoreCase(instruction) || RESUME.name().equalsIgnoreCase(instruction);
29      }
30  
31      public boolean matches(String instruction) {
32          return this.name().equalsIgnoreCase(instruction);
33      }
34  }
