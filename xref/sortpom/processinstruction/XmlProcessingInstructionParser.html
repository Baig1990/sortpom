1   package sortpom.processinstruction;
2   
3   import sortpom.exception.FailureException;
4   import sortpom.logger.SortPomLogger;
5   
6   /**
7    * Handling for xml processing instructions in the pom file. Supports ignore and resume
8    *
9    * @author bjorn
10   * @since 2013-12-28
11   */
12  public class XmlProcessingInstructionParser {
13      private final IgnoredSectionsStore ignoredSectionsStore = new IgnoredSectionsStore();
14      private String originalXml;
15      private SortPomLogger logger;
16      private boolean containsIgnoredSections = false;
17  
18      public void setup(SortPomLogger logger) {
19          this.logger = logger;
20      }
21  
22      /** Checks if pom file contains any processing instructions */
23      public void scanForIgnoredSections(String originalXml) {
24          this.originalXml = originalXml;
25          SortpomPiScanner sortpomPiScanner = new SortpomPiScanner(logger);
26          sortpomPiScanner.scan(originalXml);
27          if (sortpomPiScanner.isScanError()) {
28              throw new FailureException(sortpomPiScanner.getFirstError());
29          }
30          containsIgnoredSections = sortpomPiScanner.containsIgnoredSections();
31      }
32  
33      /** Returns true if sortpom processing instructions exists */
34      public boolean existsIgnoredSections() {
35          return containsIgnoredSections;
36      }
37  
38      /** Stores all ignored sections and replaces each with a processing instruction token */
39      public String replaceIgnoredSections() {
40          if (containsIgnoredSections) {
41              return ignoredSectionsStore.replaceIgnoredSections(originalXml);
42          }
43          return originalXml;
44      }
45  
46      /** Reverts the processing instruction token back to original content */
47      public String revertIgnoredSections(String sortedXml) {
48          if (containsIgnoredSections) {
49              return ignoredSectionsStore.revertIgnoredSections(sortedXml);
50          }
51          return sortedXml;
52      }
53  }
