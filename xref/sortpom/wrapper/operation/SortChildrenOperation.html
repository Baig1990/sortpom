6   import java.util.List;

8   /**
9    * Xml hierarchy operation that all xml elements. Used by 
10   * Used in HierarchyWrapper.processOperation(HierarchyWrapperOperation operation)
11   * @author bjorn
12   * @since 2013-11-01
13   */
14  public class SortChildrenOperation extends HierarchyWrapperOperation {
15      
16      /** Sort all children of an element */
17      @Override
18      public void manipulateChildElements(List<HierarchyWrapper> children) {
19          for (int i = 0; i < children.size(); i++) {
20              HierarchyWrapper wrapperImpl = children.get(i);
21              final Wrapper<Element> wrapper = wrapperImpl.getElementContent();
22  
23              if (wrapper != null && wrapper.isSortable()) {
24                  insertChildInSortedOrder(children, i, wrapperImpl, wrapper);
25              }
26          }
27  
28      }
29  
30      private void insertChildInSortedOrder(List<HierarchyWrapper> children, int indexOfChild, HierarchyWrapper wrapperImpl, Wrapper<Element> wrapper) {
31          boolean done = false;
32          for (int j = 0; !done && j < indexOfChild; j++) {
33              if (wrapper.isBefore(children.get(j).getElementContent())) {
34                  children.remove(indexOfChild);
35                  children.add(j, wrapperImpl);
36                  done = true;
37              }
38          }
39      }
40  }
