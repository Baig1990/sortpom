1   package sortpom.wrapper.operation;

3   import org.jdom.Content;
4   import org.jdom.Element;
5   import sortpom.wrapper.content.Wrapper;
6   
7   /**
8    * Xml hierarchy operation that detaches a xml child from its parent. Used by 
9    * Used in HierarchyWrapper.processOperation(HierarchyWrapperOperation operation)
10   * 
11   * @author bjorn
12   * @since 2013-11-01
13   */
14  public class DetachOperation extends HierarchyWrapperOperation {
15      /** Detach each 'other content' */
16      @Override
17      public void processOtherContent(Wrapper<Content> contentWrapper) {
18          contentWrapper.getContent().detach();
19      }
20  
21      /** Detach each xml element */
22      @Override
23      public void processElement(Wrapper<Element> elementWrapper) {
24          Element content = elementWrapper.getContent();
25          content.detach();
26          content.removeContent();
27      }
28  
29  }
