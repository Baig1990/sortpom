3   import org.jdom.Content;
4   import org.jdom.Element;
5   import sortpom.wrapper.content.Wrapper;

7   import java.util.List;

9   /**
10   * This class gives a default implementation of an operation that traverse the xml hierarchy.
11   * The operation executes on each xml element recursively 
12   * @author bjorn
13   * @since 2013-11-01
14   */
15  public abstract class HierarchyWrapperOperation {
16      /** Override this if the operation wants to do something before each element has been processed */
17      public void startOfProcess() {
18      }
19  
20      /** Override this if the operation wants to do something with each 'other content' that
21       * belongs to the element being processed
22       * @param content such as newlines and comments
23       */
24      public void processOtherContent(Wrapper<Content> content) {
25      }
26  
27      /** Override this if the operation wants to do something with the actual element being processed */
28      public void processElement(Wrapper<Element> element) {
29      }
30  
31      /** Override this if the operation want to manipulate the child elements of element being processed */
32      public void manipulateChildElements(List<HierarchyWrapper> children) {
33      }
34  
35      /** Override this if the operation wants to manipulate itself before it traverses down to the children
36       * of the element being processed
37       * @return the same or another instance of the implementing operation
38       */
39      public HierarchyWrapperOperation createSubOperation() {
40          return this;
41      }
42  
43      /** Override this if the operation wants to do something after each element has been processed */
44      public void endOfProcess() {
45      }
46  
47  }
