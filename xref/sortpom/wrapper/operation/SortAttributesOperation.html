3   import org.jdom.Attribute;
4   import org.jdom.Element;
5   import sortpom.wrapper.content.Wrapper;
6   
7   import java.util.ArrayList;
8   import java.util.Collections;
9   import java.util.List;
10  
11  /**
12   * Xml hierarchy operation that sort all attributes of xml elements. Used by 
13   * Used in HierarchyWrapper.processOperation(HierarchyWrapperOperation operation)
14   * @author bjorn
15   * @since 2013-11-01
16   */
17  public class SortAttributesOperation extends HierarchyWrapperOperation {
18      private static final AttributeComparator ATTRIBUTE_COMPARATOR = new AttributeComparator();
19  
20      /** Sort attributes of each element */
21      @Override
22      public void processElement(Wrapper<Element> elementWrapper) {
23          Element element = elementWrapper.getContent();
24          element.setAttributes(getSortedAttributes(element));
25      }
26  
27      private List<Attribute> getSortedAttributes(Element element) {
28          final List<Attribute> attributes = getAttributeList(element);
29          for (Attribute attribute : attributes) {
30              attribute.detach();
31          }
32          Collections.sort(attributes, ATTRIBUTE_COMPARATOR);
33          return attributes;
34      }
35  
36      @SuppressWarnings("unchecked")
37      private List<Attribute> getAttributeList(final Element element) {
38          return new ArrayList<Attribute>(element.getAttributes());
39      }
40  
41  }
