10  /**
11   * @author bjorn
12   * @since 2012-05-19
13   */
14  public class ElementWrapperCreator {
15      private DependencySortOrder sortDependencies;
16      private DependencySortOrder sortPlugins;
17      private boolean sortProperties;
18      private final ElementSortOrderMap elementNameSortOrderMap;
19  
20  
21      public ElementWrapperCreator(ElementSortOrderMap elementNameSortOrderMap) {
22          this.elementNameSortOrderMap = elementNameSortOrderMap;
23      }
24  
25      public void setup(PluginParameters pluginParameters) {
26          this.sortDependencies = pluginParameters.sortDependencies;
27          this.sortPlugins = pluginParameters.sortPlugins;
28          this.sortProperties = pluginParameters.sortProperties;
29      }
30  
31      public Wrapper<Element> createWrapper(Element element) {
32          boolean sortedBySortOrderFile = elementNameSortOrderMap.containsElement(element);
33          if (sortedBySortOrderFile) {
34              if (isDependencyElement(element)) {
35                  DependencySortedWrapper dependencySortedWrapper = new DependencySortedWrapper(element, elementNameSortOrderMap.getSortOrder(element));
36                  dependencySortedWrapper.setSortOrder(sortDependencies);
37                  return dependencySortedWrapper;
38              }
39              if (isPluginElement(element)) {
40                  PluginSortedWrapper pluginSortedWrapper = new PluginSortedWrapper(element, elementNameSortOrderMap.getSortOrder(element));
41                  pluginSortedWrapper.setSortOrder(sortPlugins);
42                  return pluginSortedWrapper;
43              }
44              return new SortedWrapper(element, elementNameSortOrderMap.getSortOrder(element));
45          }
46          if (isPropertyElement(element)) {
47              return new AlphabeticalSortedWrapper(element);
48          }
49          return new UnsortedWrapper<Element>(element);
50      }
51  
52      private boolean isDependencyElement(final Element element) {
53          if (sortDependencies.isNoSorting()) {
54              return false;
55          }
56          return isElementName(element, "dependency") && isElementParentName(element, "dependencies");
57      }
58  
59      private boolean isPluginElement(final Element element) {
60          if (sortPlugins.isNoSorting()) {
61              return false;
62          }
63          if (isElementName(element, "plugin")) {
64              return isElementParentName(element, "plugins") || isElementParentName(element, "reportPlugins");
65          }
66          return false;
67      }
68  
69      private boolean isPropertyElement(final Element element) {
70          if (!sortProperties) {
71              return false;
72          }
73          String deepName = getDeepName(element);
74          boolean inTheRightPlace = deepName.startsWith("/project/properties/")
75                  || deepName.startsWith("/project/profiles/profile/properties/");
76          return inTheRightPlace && isElementParentName(element, "properties");
77      }
